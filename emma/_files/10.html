381    /**
382     * Since Commons Logging doesn't have a first class notion of Level objects or integer values, this method
383     * converts the level value to one of the <tt>isXYZEnabled()</tt> methods on the Log.
384     *
385     * @param atLevel The level at which a log message is logged.
386     * @return true if the Log used by this StopWatch will output messages at the level specified.
387     */
388    protected boolean isLogging(int atLevel) {
389        switch (atLevel) {
390        case TRACE_LEVEL:
391            return logger.isTraceEnabled();
392        case DEBUG_LEVEL:
393            return logger.isDebugEnabled();
394        case INFO_LEVEL:
395            return logger.isInfoEnabled();
396        case WARN_LEVEL:
397            return logger.isWarnEnabled();
398        case ERROR_LEVEL:
399            return logger.isErrorEnabled();
400        case FATAL_LEVEL:
401            return logger.isFatalEnabled();
402        default:
403            // if here it means the level was set to a non-standard value.
404            // Only expend the time to find the closest known level if we need to.
405            return isLogging(closestKnownLevel(atLevel));
406        }
407    }

409    /**
410     * Since Commons Logging doesn't have a first class notion of Level objects or integer values, this method
411     * maps the atLevel parameter to one of the <tt>trace()</tt>, <tt>debug()</tt>, <tt>info()</tt>, <tt>warn()</tt>,
412     * <tt>error()</tt>, or <tt>fatal()</tt> methods on the Log.
413     *
414     * @param stopWatchAsString The stringified view of the stop watch for logging.
415     * @param exception         An exception, if any, that was passed to the stop or lap method. If this is null then
416     *                          logging will occur at normalPriority, if non-null it will occur at exceptionPriority.
417     * @param atLevel           The level at which logging should occur.
418     */
419    protected void log(String stopWatchAsString, Throwable exception, int atLevel) {
420        switch (atLevel) {
421        case TRACE_LEVEL:
422            logger.trace(stopWatchAsString, exception);
423            break;
424        case DEBUG_LEVEL:
425            logger.debug(stopWatchAsString, exception);
426            break;
427        case INFO_LEVEL:
428            logger.info(stopWatchAsString, exception);
429            break;
430        case WARN_LEVEL:
431            logger.warn(stopWatchAsString, exception);
432            break;
433        case ERROR_LEVEL:
434            logger.error(stopWatchAsString, exception);
435            break;
436        case FATAL_LEVEL:
437            logger.fatal(stopWatchAsString, exception);
438            break;
439        default:
440            // if here it means the level was set to a non-standard value.
441            // Only expend the time to find the closest known level if we need to.
442            log(stopWatchAsString, exception, closestKnownLevel(atLevel));
443        }
444    }

446    private int closestKnownLevel(int level) {
447        if (level <= TRACE_LEVEL) {
448            return TRACE_LEVEL;
449        }
450        if (level <= DEBUG_LEVEL) {
451            return DEBUG_LEVEL;
452        }
453        if (level <= INFO_LEVEL) {
454            return INFO_LEVEL;
455        }
456        if (level <= WARN_LEVEL) {
457            return WARN_LEVEL;
458        }
459        if (level <= ERROR_LEVEL) {
460            return ERROR_LEVEL;
461        }
462        return FATAL_LEVEL;
463    }

466    /**
467     * This utility method provides the standard mapping between log4j level names (which Perf4J uses as the standard
468     * set of possible levels) to the corresponding ..._LEVEL constant from this class.
469     *
470     * @param levelName The name of the logging level, should be one of TRACE, DEBUG, INFO, WARN, ERROR or FATAL.
471     * @return The corresponding ..._LEVEL constant from this class.
472     */
473    public static int mapLevelName(String levelName) {
474        levelName = levelName.toUpperCase();
475        if ("TRACE".equals(levelName)) { return TRACE_LEVEL; }
476        if ("DEBUG".equals(levelName)) { return DEBUG_LEVEL; }
477        if ("INFO".equals(levelName)) { return INFO_LEVEL; }
478        if ("WARN".equals(levelName)) { return WARN_LEVEL; }
479        if ("ERROR".equals(levelName)) { return ERROR_LEVEL; }
480        if ("FATAL".equals(levelName)) { return FATAL_LEVEL; }
481        return INFO_LEVEL;
482    }
