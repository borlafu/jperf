358    /**
359     * This utility method provides the standard mapping between log4j level names (which Perf4J uses as the standard
360     * set of possible levels) to the closest corresponding java.util.logging Level.
361     *
362     * @param levelName The name of the logging level, should be one of TRACE, DEBUG, INFO, WARN, ERROR or FATAL.
363     * @return The java.util.logging Level that is the closest match to the log4j Level name.
364     */
365    public static Level mapLevelName(String levelName) {
366        levelName = levelName.toUpperCase();
367        if ("TRACE".equals(levelName)) { return Level.FINEST; }
368        if ("DEBUG".equals(levelName)) { return Level.FINE; }
369        if ("INFO".equals(levelName)) { return Level.INFO; }
370        if ("WARN".equals(levelName)) { return Level.WARNING; }
371        if ("ERROR".equals(levelName)) { return Level.SEVERE; }
372        if ("FATAL".equals(levelName)) { return Level.SEVERE; }
373        return Level.INFO;
374    }
