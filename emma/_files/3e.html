382    /**
383     * Since Commons Logging doesn't have a first class notion of Level objects or integer values, this method
384     * converts the level value to one of the <tt>isXYZEnabled()</tt> methods on the Log.
385     *
386     * @param atLevel The level at which a log message is logged.
387     * @return true if the Logger used by this StopWatch will output messages at the level specified.
388     */
389    protected boolean isLogging(int atLevel) {
390        switch (atLevel) {
391        case TRACE_LEVEL:
392            return logger.isTraceEnabled();
393        case DEBUG_LEVEL:
394            return logger.isDebugEnabled();
395        case INFO_LEVEL:
396            return logger.isInfoEnabled();
397        case WARN_LEVEL:
398            return logger.isWarnEnabled();
399        case ERROR_LEVEL:
400            return logger.isErrorEnabled();
401        default:
402            // if here it means the level was set to a non-standard value.
403            // Only expend the time to find the closest known level if we need to.
404            return isLogging(closestKnownLevel(atLevel));
405        }
406    }

408    /**
409     * Since Commons Logging doesn't have a first class notion of Level objects or integer values, this method
410     * maps the atLevel parameter to one of the <tt>trace()</tt>, <tt>debug()</tt>, <tt>info()</tt>, <tt>warn()</tt>,
411     * <tt>error()</tt>, or <tt>fatal()</tt> methods on the Log.
412     *
413     * @param stopWatchAsString The stringified view of the stop watch for logging.
414     * @param exception         An exception, if any, that was passed to the stop or lap method. If this is null then
415     *                          logging will occur at normalPriority, if non-null it will occur at exceptionPriority.
416     * @param atLevel           The level at which logging should occur.
417     */
418    protected void log(String stopWatchAsString, Throwable exception, int atLevel) {
419        switch (atLevel) {
420        case TRACE_LEVEL:
421            logger.trace(stopWatchAsString, exception);
422            break;
423        case DEBUG_LEVEL:
424            logger.debug(stopWatchAsString, exception);
425            break;
426        case INFO_LEVEL:
427            logger.info(stopWatchAsString, exception);
428            break;
429        case WARN_LEVEL:
430            logger.warn(stopWatchAsString, exception);
431            break;
432        case ERROR_LEVEL:
433            logger.error(stopWatchAsString, exception);
434            break;
435        default:
436            // if here it means the level was set to a non-standard value.
437            // Only expend the time to find the closest known level if we need to.
438            log(stopWatchAsString, exception, closestKnownLevel(atLevel));
439        }
440    }

442    private int closestKnownLevel(int level) {
443        if (level <= TRACE_LEVEL) {
444            return TRACE_LEVEL;
445        }
446        if (level <= DEBUG_LEVEL) {
447            return DEBUG_LEVEL;
448        }
449        if (level <= INFO_LEVEL) {
450            return INFO_LEVEL;
451        }
452        if (level <= WARN_LEVEL) {
453            return WARN_LEVEL;
454        }
455        return ERROR_LEVEL;
456    }

459    /**
460     * This utility method provides the standard mapping between log4j level names (which Perf4J uses as the standard
461     * set of possible levels) to the corresponding ..._LEVEL constant from this class.
462     *
463     * @param levelName The name of the logging level, should be one of TRACE, DEBUG, INFO, WARN, ERROR or FATAL.
464     * @return The corresponding ..._LEVEL constant from this class.
465     */
466    public static int mapLevelName(String levelName) {
467        levelName = levelName.toUpperCase();
468        if ("TRACE".equals(levelName)) { return TRACE_LEVEL; }
469        if ("DEBUG".equals(levelName)) { return DEBUG_LEVEL; }
470        if ("INFO".equals(levelName)) { return INFO_LEVEL; }
471        if ("WARN".equals(levelName)) { return WARN_LEVEL; }
472        if ("ERROR".equals(levelName)) { return ERROR_LEVEL; }
473        if ("FATAL".equals(levelName)) { return ERROR_LEVEL; } //SLF4J has no FATAL level
474        return INFO_LEVEL;
475    }
