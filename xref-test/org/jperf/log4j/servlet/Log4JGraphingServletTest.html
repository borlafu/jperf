4   package org.jperf.log4j.servlet;
5   
6   import junit.framework.TestCase;
7   import org.springframework.mock.web.MockHttpServletRequest;
8   import org.springframework.mock.web.MockHttpServletResponse;
9   
10  /**
11   * Tests the log4j GraphingServlet.
12   */
13  public class Log4JGraphingServletTest extends TestCase {
14  
15      /**
16       * Test for http://jira.codehaus.org/browse/PERFFORJ-28
17       *
18       * @throws Exception Thrown on error
19       */
20      public void testUnknownGraphName() throws Exception {
21          GraphingServlet servlet = new GraphingServlet();
22  
23          MockHttpServletRequest request = new MockHttpServletRequest("GET", "/jperf");
24          MockHttpServletResponse response = new MockHttpServletResponse();
25  
26          request.addParameter("graphName", "unknownGraph");
27  
28          servlet.service(request, response);
29  
30          //we should get a message that the graph name was unknown, not an NPE
31          String content = response.getContentAsString();
32          assertTrue("Didn't find expected warning message in response: " + content,
33                     content.indexOf("Unknown graph name: unknownGraph") >= 0);
34      }
35  }
