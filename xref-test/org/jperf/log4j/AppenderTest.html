73      // http://jira.codehaus.org/browse/PERFFORJ-21
74      public void testAppendersTimesliceOver() throws Exception {
75      	// need to do immediateflush on the fileappender since close will not be called
76          DOMConfigurator.configure(getClass().getResource("log4j-timeslicebug.xml"));
77  
78          AsyncCoalescingStatisticsAppender appender =
79                  (AsyncCoalescingStatisticsAppender) Logger.getLogger(StopWatch.DEFAULT_LOGGER_NAME)
80                          .getAppender("coalescingStatistics");
81  
82          //log from a bunch of threads
83          TestLoggingThread[] testThreads = new TestLoggingThread[10];
84          for (int i = 0; i < testThreads.length; i++) {
85              testThreads[i] = new TestLoggingThread();
86              testThreads[i].start();
87          }
88  
89          for (TestLoggingThread testThread : testThreads) {
90              testThread.join();
91          }
92          
93          // we should see all the logging after waiting this long
94          Thread.sleep(2*appender.getTimeSlice());
95  
96          //simple verification ensures that the total number of logged messages is correct.
97          //tagName  avg           min     max     std dev       count, which is group 1
98          String regex = "tag\\d+\\s*\\d+\\.\\d\\s*\\d+\\s*\\d+\\s*\\d+\\.\\d\\s*(\\d+)";
99          Pattern statLinePattern = Pattern.compile(regex);
100         Scanner scanner = new Scanner(new File("target/statisticsLog-timeslicebug.log"));
101 
102         int totalCount = 0;
103         while (scanner.findWithinHorizon(statLinePattern, 0) != null) {
104             totalCount += Integer.parseInt(scanner.match().group(1));
105         }
106         assertEquals(testThreads.length * TestLoggingThread.STOP_WATCH_COUNT, totalCount);
107     }

109     //test for http://jira.codehaus.org/browse/PERFFORJ-22
110     public void testFlushOnShutdown() throws Exception {
111         DOMConfigurator.configure(getClass().getResource("log4j-shutdownbug.xml"));
112 
113         //make a bunch of logs, but not enough to go over the timeslice.
114         for (int i = 0; i < 5; i++) {
115             StopWatch stopWatch = new Log4JStopWatch("tag1");
116             Thread.sleep(10 * i);
117             stopWatch.stop();
118         }
119 
120         //at this point none of the file appenders will have written anything because they haven't been flushed
121         assertEquals("", FileUtils.readFileToString(new File("target/stats-shutdownbug.log")));
122         assertEquals("", FileUtils.readFileToString(new File("target/graphs-shutdownbug.log")));
123 
124         //now, to simulate shutdown, get the async appender and run the shutdown hook. We need to use reflection
125         //because the shutdown hook is private.
126         AsyncCoalescingStatisticsAppender appender =
127                 (AsyncCoalescingStatisticsAppender) Logger.getLogger(StopWatch.DEFAULT_LOGGER_NAME)
128                         .getAppender("coalescingStatistics");
129         Field shutdownField = appender.getClass().getDeclaredField("shutdownHook");
130         shutdownField.setAccessible(true);
131         AsyncCoalescingStatisticsAppender.ShutdownHook shutdownHook = (AsyncCoalescingStatisticsAppender.ShutdownHook) shutdownField.get(appender);
132         assertNotNull("shutdownHook should not be null", shutdownHook);
133         assertNotNull("shutdownHook appender should not be null", shutdownHook.getAppender());
134         shutdownHook.run();
135 
136         //now there should be data in the files
137         assertFalse("".equals(FileUtils.readFileToString(new File("target/stats-shutdownbug.log"))));
138         assertFalse("".equals(FileUtils.readFileToString(new File("target/graphs-shutdownbug.log"))));
139 
140         // closing should remove shutdown hook
141         appender.close();
142 
143         // hard to test hook removal, but we set it to null at the same time, so we'll rely on that
144         shutdownHook = (AsyncCoalescingStatisticsAppender.ShutdownHook) shutdownField.get(appender);
145         assertNull("shutdownHook should be null", shutdownHook);
146     }

148     public void testOverflowHandling() throws Exception {
149         Logger logger = Logger.getLogger("AppenderTest.overflowTest");
150         AsyncCoalescingStatisticsAppender appender = new AsyncCoalescingStatisticsAppender();
151         appender.setName("overflowTestAppender");
152         appender.setTimeSlice(1000);
153         appender.setQueueSize(2); //set low queue size so we overflow
154         logger.addAppender(appender);
155         logger.setAdditivity(false);
156         logger.setLevel(Level.INFO);
157         appender.activateOptions();
158 
159         for (int i = 0; i < 1000; i++) {
160             StopWatch stopWatch = new StopWatch("testOverflow");
161             Math.random(); //this should happen super fast, faster than the appender's Dispatcher thread can drain
162             logger.info(stopWatch.stop());
163         }
164 
165         //again close the appender
166         appender.close();
167         assertTrue("Expected some stop watch messages to get discarded", appender.getNumDiscardedMessages() > 0);
168     }

170     public void testCsvRenderer() throws Exception {
171         DOMConfigurator.configure(getClass().getResource("log4jWCsv.xml"));
172 
173         Logger logger = Logger.getLogger("org.jperf.CsvAppenderTest");
174 
175         for (int i = 0; i < 20; i++) {
176             StopWatch stopWatch = new Log4JStopWatch(logger);
177             Thread.sleep(i * 10);
178             stopWatch.stop("csvTest");
179         }
180 
181         //close the appender
182         logger.getAppender("coalescingStatistics").close();
183 
184         //verify the statisticsLog.csv file
185         DateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
186         for (Object line : FileUtils.readLines(new File("./target/statisticsLog.csv"))) {
187             String[] values = line.toString().split(",");
188             //first column is the tag
189             assertEquals("\"csvTest\"", values[0]);
190             //next 2 columns are the dates - ensure they can be parsed
191             assertTrue(dateFormat.parse(values[1]).before(dateFormat.parse(values[2])));
192             //next 3 columns are mean, min and max
193             double mean = Double.parseDouble(values[3]);
194             int min = Integer.parseInt(values[4]);
195             int max = Integer.parseInt(values[5]);
196             assertTrue(mean >= min && mean <= max);
197             //next column is stddev - ust make sure it's parseable
198             Double.parseDouble(values[6]);
199             //next column is count
200             assertTrue(Integer.parseInt(values[7]) < 20);
201             //final column is TPS - just make sure it's parseable
202             Double.parseDouble(values[8]);
203         }
204 
205         //verify the pivotedStatisticsLog.csv file
206         for (Object line : FileUtils.readLines(new File("./target/pivotedStatisticsLog.csv"))) {
207             String[] values = line.toString().split(",");
208             //first 2 columns are the dates - ensure they can be parsed
209             assertTrue(dateFormat.parse(values[0]).before(dateFormat.parse(values[1])));
210             //next column is mean, ensure it can be parsed
211             Double.parseDouble(values[2]);
212             //last column should be empty, so make sure last char on string is comma
213             assertEquals(',', line.toString().charAt(line.toString().length() - 1));
214         }

217     protected static class TestLoggingThread extends Thread {
218         protected static final AtomicInteger index = new AtomicInteger();
219 
220         public static final int STOP_WATCH_COUNT = 20;
221 
222         public void run() {
223             String loggingTag = "tag" + index.getAndIncrement();
224 
225             for (int i = 0; i < STOP_WATCH_COUNT; i++) {
226                 StopWatch stopWatch = new Log4JStopWatch(loggingTag);
227                 int sleepTime = (int) (Math.random() * 400);
228                 try { Thread.sleep(sleepTime); } catch (Exception e) { /*do nothing*/ }
229                 stopWatch.stop();
230             }
231 
232             //this last sleep here seems necessary because otherwise Thread.join() above is returning before
233             //we're really done.
234             try { Thread.sleep(10); } catch (Exception e) { /*ignore*/ }
235         }
236     }
