16  package org.jperf.helpers;
17  
18  import org.jperf.GroupedTimingStatistics;
19  import org.jperf.TimingTestCase;
20  import org.jperf.StopWatch;
21  
22  import javax.management.MBeanAttributeInfo;
23  import javax.management.MBeanInfo;
24  import javax.management.NotificationListener;
25  import javax.management.Notification;
26  import java.util.Arrays;
27  
28  /**
29   * Tests the StatisticsExposingMBean.
30   */
31  public class StatisticsExposingMBeanTest extends TimingTestCase {
32  
33      public void testStatisticsExposingMBean() throws Exception {
34          GroupedTimingStatistics groupedTimingStats = new GroupedTimingStatistics();
35          groupedTimingStats.setStartTime(System.currentTimeMillis());
36          groupedTimingStats.setStopTime(System.currentTimeMillis() + 1000L);
37          groupedTimingStats.addStopWatches(this.testStopWatches);
38  
39          StatisticsExposingMBean mBean = new StatisticsExposingMBean(StatisticsExposingMBean.DEFAULT_MBEAN_NAME,
40                                                                      Arrays.asList("tag", "tag3"),
41                                                                      null /* no notifications */);
42          mBean.updateCurrentTimingStatistics(groupedTimingStats);
43  
44          MBeanInfo mBeanInfo = mBean.getMBeanInfo();
45          MBeanAttributeInfo[] attributeInfos = mBeanInfo.getAttributes();
46          assertEquals(mBean.getStatsValueRetrievers().size() * 2, attributeInfos.length);
47  
48          assertEquals(groupedTimingStats.getStatisticsByTag().get("tag").getMean(),
49                       mBean.getAttribute("tagMean"));
50          assertEquals(groupedTimingStats.getStatisticsByTag().get("tag").getStandardDeviation(),
51                       mBean.getAttribute("tagStdDev"));
52          assertEquals(groupedTimingStats.getStatisticsByTag().get("tag").getMax(),
53                       mBean.getAttribute("tagMax"));
54          assertEquals(groupedTimingStats.getStatisticsByTag().get("tag").getMin(),
55                       mBean.getAttribute("tagMin"));
56          assertEquals(groupedTimingStats.getStatisticsByTag().get("tag").getCount(),
57                       mBean.getAttribute("tagCount"));
58          assertEquals(((double) groupedTimingStats.getStatisticsByTag().get("tag").getCount()) /
59                       ((double) (groupedTimingStats.getStopTime() - groupedTimingStats.getStartTime()) / 1000.0),
60                       mBean.getAttribute("tagTPS"));
61  
62          //test notifications
63          DummyNotificationListener notificationListener = new DummyNotificationListener();
64          mBean = new StatisticsExposingMBean(StatisticsExposingMBean.DEFAULT_MBEAN_NAME,
65                                              Arrays.asList("tag", "tag3"),
66                                              Arrays.asList(new AcceptableRangeConfiguration("tagMean(<2000)"),
67                                                            new AcceptableRangeConfiguration("unknownTagMean(<100)")));
68          mBean.addNotificationListener(notificationListener, null, null);
69  
70          //no notification should be sent here, the timing stats are good
71          mBean.updateCurrentTimingStatistics(groupedTimingStats);
72          Thread.sleep(50); //need to sleep because notification is sent in a separate thread.
73          assertNull(notificationListener.lastReceivedNotification);
74  
75          //add a stop watch that should make the mean be too high, causing a notification to be sent
76          GroupedTimingStatistics badStats = groupedTimingStats.clone();
77          badStats.addStopWatch(new StopWatch(groupedTimingStats.getStartTime(), 100000L, "tag", "message"));
78          mBean.updateCurrentTimingStatistics(badStats);
79          Thread.sleep(50);
80          assertEquals(StatisticsExposingMBean.OUT_OF_RANGE_NOTIFICATION_TYPE,
81                       notificationListener.lastReceivedNotification.getType());
82          notificationListener.lastReceivedNotification = null;
83  
84          //now when we update the timing stats again we should NOT receive another notification, because
85          //we only want a notification the first time it goes bad.
86          mBean.updateCurrentTimingStatistics(badStats);
87          Thread.sleep(50);
88          assertNull(notificationListener.lastReceivedNotification);
89  
90          //one more time go from good to bad - we should get another notification
91          mBean.updateCurrentTimingStatistics(groupedTimingStats);
92          Thread.sleep(50);
93          assertNull(notificationListener.lastReceivedNotification);
94          mBean.updateCurrentTimingStatistics(badStats);
95          Thread.sleep(50);
96          assertEquals(StatisticsExposingMBean.OUT_OF_RANGE_NOTIFICATION_TYPE,
97                       notificationListener.lastReceivedNotification.getType());
98          notificationListener.lastReceivedNotification = null;
99  
100         //test invalid AcceptableRangeConfiguration string
101         try {
102             new StatisticsExposingMBean(StatisticsExposingMBean.DEFAULT_MBEAN_NAME,
103                                         Arrays.asList("tag"),
104                                         Arrays.asList(new AcceptableRangeConfiguration("tagNoStat(<100)")));
105             fail("Should have thrown an illegal argument exception");
106         } catch (IllegalArgumentException iae) { /* expected */ }
107         
108         //TODO - more tests - update current statistics, check for unsupported ops.
109     }
110 
111     protected static class DummyNotificationListener implements NotificationListener {
112         public Notification lastReceivedNotification;
113         
114         public void handleNotification(Notification notification, Object handback) {
115             lastReceivedNotification = notification;
116         }
117     }
118 }
