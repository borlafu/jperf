17  package net.jperf.aop;
18  
19  import net.jperf.LoggingStopWatch;
20  import net.jperf.helpers.JperfProperties;
21  import org.junit.After;
22  import org.junit.Before;
23  import org.junit.Test;
24  
25  import static org.junit.Assert.*;
26  
27  /**
28   * @author <a href="mailto:oscar.eriksson@sigma.se">Oscar Eriksson</a>
29   * @date 2/13/15
30   */
31  public class AgnosticTimingAspectThresholdTest {
32      private AgnosticTimingAspect agnosticTimingAspect;
33      private AbstractJoinPoint joinPoint;
34      private TestObject testObject;
35  
36      @After
37      public void tearDown() {
38          InMemoryTimingAspect.logStrings.clear();
39      }
40  
41      @Before
42      public void setup() {
43          JperfProperties.INSTANCE.remove(Profiled.THRESHOLD_FIELD_NAME);
44          testObject = new TestObject();
45          agnosticTimingAspect = new AgnosticTimingAspect();
46          joinPoint = new AbstractJoinPoint() {
47              public Object proceed() throws Throwable {
48                  return null;
49              }
50  
51              public Object getExecutingObject() {
52                  return new Object();
53              }
54  
55              public Object[] getParameters() {
56                  return new Object[0];
57              }
58  
59              public String getMethodName() {
60                  return "methodName";
61              }
62  
63              public Class<?> getDeclaringClass() {
64                  return AgnosticTimingAspectTest.class;
65              }
66          };
67      }
68  
69      @Test
70      public void testStopWatchTimeThresholdDefaultValueIsZero() throws Throwable {
71          LoggingStopWatch stopWatch = new LoggingStopWatch();
72          agnosticTimingAspect.runProfiledMethod(joinPoint, DefaultProfiled.INSTANCE, stopWatch);
73          assertEquals(0, stopWatch.getTimeThreshold());
74      }
75  
76      @Test
77      public void testProfiledTimeThresholdDefaultValueIsMinusOne() throws Throwable {
78          assertEquals(-1, DefaultProfiled.INSTANCE.timeThreshold());
79      }
80  
81      @Test
82      public void testProfiledAnnotationTimeThresholdDefaultValueIsMinusOne() throws Throwable {
83          long defaultValueOnAnnotation = (Long)testObject.getClass().getMethod("annotation").getDeclaredAnnotations()[0]
84                  .annotationType().getMethod(Profiled.THRESHOLD_FIELD_NAME).getDefaultValue();
85          assertEquals(-1, defaultValueOnAnnotation);
86      }
87  
88      @Test
89      public void testThatProfiledAnnotationDefaultTimeThresholdValueIsMinusOne() throws Exception {
90          testObject.noTimeThresholdSetOnAnnotation();
91          assertNotNull("should have logged since default timeout is 0ms", InMemoryTimingAspect.getLastLoggedString());
92      }
93  
94      @Test
95      public void testDefaultThresholdWillLogMessageWithNoSleepWillLog() throws Exception {
96          testObject.noTimeThresholdSetOnAnnotation();
97          assertNotNull("should have logged since default timeout is 0ms", InMemoryTimingAspect.getLastLoggedString());
98      }
99  
100     @Test
101     public void testDefaultThresholdWillLogMessageWith100MsSleepWillLog() throws Exception {
102         testObject.noTimeThresholdSetOnAnnotationSleep100Ms();
103         assertNotNull("should have logged since default timeout is 0ms", InMemoryTimingAspect.getLastLoggedString());
104     }
105 
106     @Test
107     public void test50MsThresholdOnAnnotationWith100MsSleepWillLog() throws Exception {
108         testObject.timeThresholdOf50MsSetOnAnnotationSleep100Ms();
109         assertNotNull("should have logged since sleep is longer than set threshold", InMemoryTimingAspect.getLastLoggedString());
110     }
111 
112     @Test
113     public void tes100MsThresholdOnAnnotationWithNoSleepWillNotLog() throws Exception {
114         testObject.timeThresholdOf100MsSetOnAnnotationNoSleep();
115         assertNull("should not have logged since sleep is shorter than set threshold", InMemoryTimingAspect.getLastLoggedString());
116     }
117 
118     @Test
119     public void testConfiguredThresholdSurpessesLogging() {
120         JperfProperties.INSTANCE.setProperty(Profiled.THRESHOLD_FIELD_NAME, "2000");
121         testObject.noTimeThresholdSetOnAnnotationSleep100Ms();
122         assertNull("should not have logged", InMemoryTimingAspect.getLastLoggedString());
123     }
124 
125     @Test
126     public void testConfiguredThresholdDoesNotSurpessesLogging() {
127         JperfProperties.INSTANCE.setProperty(Profiled.THRESHOLD_FIELD_NAME, "10");
128         testObject.noTimeThresholdSetOnAnnotationSleep100Ms();
129         assertNotNull("should have logged", InMemoryTimingAspect.getLastLoggedString());
130     }
131 
132     @Test
133     public void testNoConfiguredThresholdDoesNotSurpessesLogging() {
134         testObject.noTimeThresholdSetOnAnnotationSleep100Ms();
135         assertNotNull("should have logged", InMemoryTimingAspect.getLastLoggedString());
136     }
137 
138     private static class TestObject {
139         @Profiled
140         public void annotation() throws IllegalAccessException {
141             throw new IllegalAccessException("should not be executed, only for retrieving annotation");
142         }
143 
144         @Profiled
145         public void noTimeThresholdSetOnAnnotation() {
146             System.out.println("noTimeThresholdSetOnAnnotation doing nothing");
147         }
148 
149         @Profiled
150         public void noTimeThresholdSetOnAnnotationSleep100Ms() {
151             sleep(100);
152         }
153 
154         @Profiled(timeThreshold = 50)
155         public void timeThresholdOf50MsSetOnAnnotationSleep100Ms() {
156             sleep(100);
157         }
158 
159         @Profiled(timeThreshold = 100)
160         public void timeThresholdOf100MsSetOnAnnotationNoSleep() {
161             System.out.println("timeThresholdOf100MsSetOnAnnotationNoSleep doing nothing");
162         }
163 
164         private void sleep(long ms) {
165             try {
166                 Thread.sleep(ms);
167             }
168             catch (InterruptedException e) {
169                 // nothing
170             }
171         }
172     }
173 }
