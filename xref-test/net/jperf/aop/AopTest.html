147     public void testConcurrentCalls() throws Exception {
148         InMemoryTimingAspect.logStrings.clear();
149         final List<String> expectedTags = Collections.synchronizedList(new ArrayList<String>(2000));
150 
151         //run a bunch of threads concurrently that call a @Profiled method
152         Thread[] threads = new Thread[20];
153         for (int i = 0; i < threads.length; i++) {
154             threads[i] = new Thread() {
155                 public void run() {
156                     for (int i = 0; i < 100; i++) {
157                         try {
158                             profiledObject.simpleTestWithJexlTag(i, new ProfiledObject.SimpleBean("Alex", 10));
159                             expectedTags.add("expressionTest_" + i + "_Alex_10");
160                         } catch (Exception e) {
161                             throw new RuntimeException(e);
162                         }
163                     }
164                 }
165             };
166             threads[i].start();
167         }
168 
169         //wait for all the threads to finish
170         for (Thread thread : threads) {
171             thread.join();
172         }
173         Thread.sleep(100);
174 
175         //ensure that that all of the expected tags are present in the InMemoryTimingAspect.logStrings
176         assertEquals(expectedTags.size(), InMemoryTimingAspect.logStrings.size());
177 outer:  for (String expectedTag : expectedTags) {
178             for (String logString : InMemoryTimingAspect.logStrings) {
179                 if (logString.indexOf(expectedTag) >= 0) {
180                     continue outer;
181                 }
182             }
183 
184             //if this far, expected tag wasn't found in logStrings
185             fail("Expected tag " + expectedTag + " not found");
186         }
187     }

189     /**
190      * Tests that call joinpoints are ignored when using Perf4j's AOP
191      *
192      * @throws Exception if anything bad happens
193      */
194     public void testCallsDoNotCreateDuplicateLogEntries() throws Exception {
195         //calling the profiled method should result in one log
196         InMemoryTimingAspect.logStrings.clear(); //clear the log entries
197         profiledObject.simpleTest(50);
198         assertEquals(1, InMemoryTimingAspect.logStrings.size());
199 
200         //calling the method that CALLS the profiled method should still result in only one call
201         InMemoryTimingAspect.logStrings.clear(); //clear the log entries
202         profiledObject.simpleMethodCallExample(50);
203         assertEquals(1, InMemoryTimingAspect.logStrings.size());
204     }
