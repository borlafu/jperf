17  package net.jperf;
18  
19  import java.util.SortedMap;
20  import java.util.TreeMap;
21  
22  import junit.framework.TestCase;
23  
24  /**
25   * PERFFORJ-55.
26   */
27  public class GroupedTimingStatisticsTest extends TestCase {
28  
29      private final boolean createRollupStatistics = false;
30      private final long startTime = System.currentTimeMillis();
31      private final long stopTime = startTime + 100;
32      private final SortedMap<String, TimingStatistics> statisticsByTag = new TreeMap<String, TimingStatistics>();
33  
34      public void testOutputHasPaddingForTagHeadingWithEmptyGroupedTimingStatistics() throws Exception {
35          GroupedTimingStatistics groupStatistics = new GroupedTimingStatistics(statisticsByTag, startTime, stopTime, createRollupStatistics);
36          assertOutputContains(groupStatistics.toString(), "Tag     Avg(ms)         Min         Max     Std-Dev       Count");
37      }
38  
39      public void testOutputHasPaddingAllowingForTheLongestTag() throws Exception {
40          statisticsByTag.put("a short tag", new TimingStatistics(0.1, 0.2, 3, 4, 5));
41          statisticsByTag.put("a very very very very very very very very very very very very long tag",
42                  new TimingStatistics(0.1, 0.2, 3, 4, 5));
43          statisticsByTag.put("another short tag", new TimingStatistics(0.1, 0.2, 3, 4, 5));
44  
45          GroupedTimingStatistics groupStatistics = new GroupedTimingStatistics(statisticsByTag, startTime, stopTime, createRollupStatistics);
46          String output = groupStatistics.toString();
47  
48          assertOutputContains(output, "Tag                                                                        Avg(ms)         Min         Max     Std-Dev       Count");
49          assertOutputContains(output, "a short tag                                                                    0.1           4           3         0.2           5");
50          assertOutputContains(output, "a very very very very very very very very very very very very long tag         0.1           4           3         0.2           5");
51          assertOutputContains(output, "another short tag                                                              0.1           4           3         0.2           5");
52      }
53  
54      public void testOutputContainsHeadingForTotalTimeSpentPerTag() throws Exception {
55          GroupedTimingStatistics groupStatistics = new GroupedTimingStatistics(statisticsByTag, startTime, stopTime, createRollupStatistics);
56          assertOutputContains(groupStatistics.toString(), "Tag     Avg(ms)         Min         Max     Std-Dev       Count       Total");
57      }
58  
59      public void testOutputContainsTotalTimeSpentPerTagAsMeanxCountRoundedToZeroDecimalPlaces() throws Exception {
60          statisticsByTag.put("a", new TimingStatistics(1.1, 0.2, 3, 4, 5));
61  
62          GroupedTimingStatistics groupStatistics = new GroupedTimingStatistics(statisticsByTag, startTime, stopTime, createRollupStatistics);
63          System.out.println(groupStatistics.toString());
64          assertOutputContains(groupStatistics.toString(), "Tag     Avg(ms)         Min         Max     Std-Dev       Count       Total");
65          assertOutputContains(groupStatistics.toString(), "a           1.1           4           3         0.2           5           6");
66      }
67  
68  
69  
70  
71      private void assertOutputContains(String output, String expectedToContain) {
72          String message = "Expected toString() output to contain the given string, matching formatting.\n" + expectedToContain +
73          "\nActual output:\n" + output;
74  
75          assertTrue(message, output.contains(expectedToContain));
76      }
77  }
