17  package net.jperf.log4j.servlet;
18  
19  import junit.framework.TestCase;
20  import org.springframework.mock.web.MockHttpServletRequest;
21  import org.springframework.mock.web.MockHttpServletResponse;
22  
23  /**
24   * Tests the log4j GraphingServlet.
25   */
26  public class Log4JGraphingServletTest extends TestCase {
27  
28      /**
29       * Test for http://jira.codehaus.org/browse/PERFFORJ-28
30       *
31       * @throws Exception Thrown on error
32       */
33      public void testUnknownGraphName() throws Exception {
34          GraphingServlet servlet = new GraphingServlet();
35  
36          MockHttpServletRequest request = new MockHttpServletRequest("GET", "/jperf");
37          MockHttpServletResponse response = new MockHttpServletResponse();
38  
39          request.addParameter("graphName", "unknownGraph");
40  
41          servlet.service(request, response);
42  
43          //we should get a message that the graph name was unknown, not an NPE
44          String content = response.getContentAsString();
45          assertTrue("Didn't find expected warning message in response: " + content,
46                     content.indexOf("Unknown graph name: unknownGraph") >= 0);
47      }
48  }
