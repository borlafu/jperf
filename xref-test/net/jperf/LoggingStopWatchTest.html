17  package net.jperf;

19  import junit.framework.TestCase;
20  import org.apache.commons.io.output.ByteArrayOutputStream;
21  
22  import java.io.ByteArrayInputStream;
23  import java.io.ObjectInputStream;
24  import java.io.ObjectOutputStream;
25  import java.io.PrintStream;

27  /**
28   * Tests the LoggingStopWatch. Can also be used to test subclasses if the loggers are set up to log to stderr.
29   */
30  public class LoggingStopWatchTest extends TestCase {
31  
32      protected PrintStream realErr;
33      protected ByteArrayOutputStream fakeErr;
34  
35      protected void setUp() throws Exception {
36          realErr = System.err;
37          fakeErr = new ByteArrayOutputStream();
38          System.setErr(new PrintStream(fakeErr, true /*autoflush*/));
39      }
40  
41      protected void tearDown() throws Exception {
42          System.setErr(realErr);
43      }
44  
45      /**
46       * Tests the StopWatch. This method is designed in such a way that it should be used to test subclasses as well
47       * if createStopWatch and checkProperties are properly implemented.
48       */
49      public void testStopWatch() throws Exception {
50          checkProperties(createStopWatch(null, null, null, null, null),
51                          StopWatch.DEFAULT_LOGGER_NAME, "INFO", "WARN", "", null);
52          checkProperties(createStopWatch(null, null, null, "tag", null),
53                          StopWatch.DEFAULT_LOGGER_NAME, "INFO", "WARN", "tag", null);
54          checkProperties(createStopWatch(null, null, null, "tag2", "message2"),
55                          StopWatch.DEFAULT_LOGGER_NAME, "INFO", "WARN", "tag2", "message2");
56          checkProperties(createStopWatch("net.jperf.TestLogger", null, null, null, null),
57                          "net.jperf.TestLogger", "INFO", "WARN", "", null);
58          checkProperties(createStopWatch("net.jperf.TestLogger", "DEBUG", null, null, null),
59                          "net.jperf.TestLogger", "DEBUG", "WARN", "", null);
60          checkProperties(createStopWatch("net.jperf.TestLogger", "DEBUG", "ERROR", null, null),
61                          "net.jperf.TestLogger", "DEBUG", "ERROR", "", null);
62          checkProperties(createStopWatch("net.jperf.TestLogger", null, null, "taga", null),
63                          "net.jperf.TestLogger", "INFO", "WARN", "taga", null);
64          checkProperties(createStopWatch("net.jperf.TestLogger", "DEBUG", null, "tagb", null),
65                          "net.jperf.TestLogger", "DEBUG", "WARN", "tagb", null);
66          checkProperties(createStopWatch("net.jperf.TestLogger", "DEBUG", "ERROR", "tagc", null),
67                          "net.jperf.TestLogger", "DEBUG", "ERROR", "tagc", null);
68          checkProperties(createStopWatch("net.jperf.TestLogger", null, null, "taga", "m1"),
69                          "net.jperf.TestLogger", "INFO", "WARN", "taga", "m1");
70          checkProperties(createStopWatch("net.jperf.TestLogger", "DEBUG", null, "tagb", "m2"),
71                          "net.jperf.TestLogger", "DEBUG", "WARN", "tagb", "m2");
72          checkProperties(createStopWatch("net.jperf.TestLogger", "DEBUG", "FATAL", "tagc", "m3"),
73                          "net.jperf.TestLogger", "DEBUG", "FATAL", "tagc", "m3");
74  
75          checkSerializationAndCloning(createStopWatch("net.jperf.TestLogger", "DEBUG", "WARN", "tagd", "m4"));
76  
77          LoggingStopWatch stopWatch = createStopWatch(null, null, null, "tag", "messageText");
78          stopWatch.stop();
79          checkExpectedLogWritten("message[messageText]");
80          stopWatch.stop("foo");
81          checkExpectedLogWritten("tag[foo]");
82          stopWatch.stop("you", "bar");
83          checkExpectedLogWritten("tag[you] message[bar]");
84          stopWatch.stop(new Exception("zoo"));
85          checkExpectedLogWritten("java.lang.Exception: zoo");
86          stopWatch.stop("blue", new Exception("goo"));
87          checkExpectedLogWritten("tag[blue]", "java.lang.Exception: goo");
88          stopWatch.stop("loo", "car", new Exception("moo"));
89          checkExpectedLogWritten("tag[loo] message[car]", "java.lang.Exception: moo");
90          stopWatch.lap("coo");
91          checkExpectedLogWritten("tag[coo]");
92          stopWatch.lap("ewe", "far");
93          checkExpectedLogWritten("tag[ewe] message[far]");
94          stopWatch.lap("poo", new Exception("shoe"));
95          checkExpectedLogWritten("tag[poo]", "java.lang.Exception: shoe");
96          stopWatch.lap("new", "mar", new Exception("rue"));
97          checkExpectedLogWritten("tag[new] message[mar]", "java.lang.Exception: rue");
98  
99          //test for PERFFORJ-30 - Add capability to set a time threshold in LoggingStopWatch and Profiled annotation
100         stopWatch.stop();
101         String fakeErrBefore = fakeErr.toString();
102         stopWatch.setTimeThreshold(100).start("timeThresholdCheck");
103         Thread.sleep(10);
104         stopWatch.stop();
105         assertEquals("Stopwatch log was set when it shouldn't have been", fakeErrBefore, fakeErr.toString());
106         //now it should get logged after running
107         stopWatch.start();
108         Thread.sleep(110);
109         stopWatch.stop();
110         checkExpectedLogWritten("tag[timeThresholdCheck]");
111 
112         customTests();
113     }
114 
115     /**
116      * Subclasses should override this to test items specific to the subclass of LoggingStopWatch.
117      */
118     protected void customTests() {
119         LoggingStopWatch stopWatch = new LoggingStopWatch(0L, 1000L, "tag", "message");
120         assertEquals(0L, stopWatch.getStartTime());
121         assertEquals(1000L, stopWatch.getElapsedTime());
122         assertEquals("tag", stopWatch.getTag());
123         assertEquals("message", stopWatch.getMessage());
124 
125         assertTrue(stopWatch.isLogging());
126     }
127 
128     /**
129      * This method is meant to be overridden in such a way that test subclasses return the proper LoggingStopWatch
130      * subclass implementation.
131      */
132     protected LoggingStopWatch createStopWatch(String loggerName,
133                                                String normalPriorityName,
134                                                String exceptionPriorityName,
135                                                String tag,
136                                                String message) {
137         //main LoggingStopWatch class doesn't care about logger or priorities
138         if (tag == null && message == null) {
139             return new LoggingStopWatch();
140         }
141         if (message == null) {
142             return new LoggingStopWatch(tag);
143         }
144         return new LoggingStopWatch(tag, message);
145     }
146 
147     /**
148      * Test subclasses should override this method to validate the additional properties.
149      */
150     protected void checkProperties(LoggingStopWatch stopWatch,
151                                    String expectedLoggerName,
152                                    String expectedNormalPriority,
153                                    String expectedExceptionPriority,
154                                    String expectedTag,
155                                    String expectedMessage) {
156         assertEquals(expectedTag, stopWatch.getTag());
157         assertEquals(expectedMessage, stopWatch.getMessage());
158     }
159 
160     protected void checkSerializationAndCloning(LoggingStopWatch stopWatch) {
161         LoggingStopWatch cloneCopy = stopWatch.clone();
162         assertEquals(stopWatch, cloneCopy);
163 
164         try {
165             ByteArrayOutputStream os = new ByteArrayOutputStream();
166             new ObjectOutputStream(os).writeObject(stopWatch);
167             LoggingStopWatch serializedCopy =
168                     (LoggingStopWatch) new ObjectInputStream(new ByteArrayInputStream(os.toByteArray())).readObject();
169             assertEquals(stopWatch, serializedCopy);
170             assertEquals(cloneCopy, serializedCopy);
171         } catch (Exception e) {
172             fail("Unexpected Exception: " + e);
173         }
174     }
175 
176     protected void checkExpectedLogWritten(String... textToFind) {
177         String fakeErrLog = fakeErr.toString();
178         for (String text : textToFind) {
179             assertTrue("'" + text + "' not found in log " + fakeErrLog, fakeErrLog.indexOf(text) >= 0);
180         }
181         fakeErr.reset();
182     }
