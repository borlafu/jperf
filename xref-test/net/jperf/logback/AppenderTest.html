40  public class AppenderTest extends TestCase {
41      public void testAppenders() throws Exception {
42          LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
43          JoranConfigurator configurator = new JoranConfigurator();
44          configurator.setContext(lc);
45          // the context was probably already configured by default configuration 
46          // rules
47          lc.reset();
48          configurator.doConfigure(getClass().getResource("logback.xml"));
49  
50          AsyncCoalescingStatisticsAppender appender = (AsyncCoalescingStatisticsAppender) 
51              lc.getLogger(StopWatch.DEFAULT_LOGGER_NAME).getAppender("coalescingStatistics");
52  
53          //log from a bunch of threads
54          TestLoggingThread[] testThreads = new TestLoggingThread[10];
55          for (int i = 0; i < testThreads.length; i++) {
56              testThreads[i] = new TestLoggingThread();
57              testThreads[i].start();
58          }
59  
60          for (TestLoggingThread testThread : testThreads) {
61              testThread.join();
62          }
63  
64          //close the output appender, which prevents us from returning until this method completes.
65          appender.stop();
66  
67          //simple verification ensures that the total number of logged messages is correct.
68          //tagName  avg           min     max     std dev       count, which is group 1
69          String regex = "tag\\d\\s*\\d+\\.\\d\\s*\\d+\\s*\\d+\\s*\\d+\\.\\d\\s*(\\d+)";
70          Pattern statLinePattern = Pattern.compile(regex);
71          Scanner scanner = new Scanner(new File("target/statisticsLogback.log"));
72  
73          int totalCount = 0;
74          while (scanner.findWithinHorizon(statLinePattern, 0) != null) {
75              totalCount += Integer.parseInt(scanner.match().group(1));
76          }
77          assertEquals(testThreads.length * TestLoggingThread.STOP_WATCH_COUNT, totalCount);
78      }

80      public void testOverflowHandling() throws Exception {
81          LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
82          Logger logger = lc.getLogger("AppenderTest.overflowTest");
83          AsyncCoalescingStatisticsAppender appender = new AsyncCoalescingStatisticsAppender();
84          appender.setName("overflowTestAppender");
85          appender.setTimeSlice(1000);
86          appender.setQueueSize(2); //set low queue size so we overflow
87          logger.addAppender(appender);
88          logger.setAdditive(false);
89          logger.setLevel(Level.INFO);
90          appender.start();
91  
92          for (int i = 0; i < 1000; i++) {
93              StopWatch stopWatch = new StopWatch("testOverflow");
94              Math.random(); //this should happen super fast, faster than the appender's Dispatcher thread can drain
95              logger.info(stopWatch.stop());
96          }
97  
98          //again close the appender
99          appender.stop();
100         assertTrue("Expected some stop watch messages to get discarded", appender.getNumDiscardedMessages() > 0);
101     }

103     public void testCsvRenderer() throws Exception {
104         LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
105         JoranConfigurator configurator = new JoranConfigurator();
106         configurator.setContext(lc);
107         // the context was probably already configured by default configuration 
108         // rules
109         lc.reset();
110         configurator.doConfigure(getClass().getResource("logbackWCsv.xml"));
111 
112         Logger logger = lc.getLogger("net.jperf.CsvAppenderTest");
113 
114         for (int i = 0; i < 20; i++) {
115             StopWatch stopWatch = new Slf4JStopWatch(logger);
116             Thread.sleep(i * 10);
117             stopWatch.stop("csvTest");
118         }
119 
120         //close the appender
121         logger.getAppender("coalescingStatistics").stop();
122 
123         //verify the statisticsLog.csv file
124         DateFormat dateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
125         for (Object line : FileUtils.readLines(new File(
126                 "./target/statisticsLogback.csv"))) {
127             String[] values = line.toString().split(",");
128             //first column is the tag
129             assertEquals("\"csvTest\"", values[0]);
130             //next 2 columns are the dates - ensure they can be parsed
131             assertTrue(dateFormat.parse(values[1]).before(dateFormat.parse(values[2])));
132             //next 3 columns are mean, min and max
133             double mean = Double.parseDouble(values[3]);
134             int min = Integer.parseInt(values[4]);
135             int max = Integer.parseInt(values[5]);
136             assertTrue(mean >= min && mean <= max);
137             //next column is stddev - ust make sure it's parseable
138             Double.parseDouble(values[6]);
139             //next column is count
140             assertTrue(Integer.parseInt(values[7]) < 20);
141             //final column is TPS - just make sure it's parseable
142             Double.parseDouble(values[8]);
143         }
144 
145         //verify the pivotedStatisticsLog.csv file
146         for (Object line : FileUtils.readLines(new File("./target/pivotedStatisticsLog.csv"))) {
147             String[] values = line.toString().split(",");
148             //first 2 columns are the dates - ensure they can be parsed
149             assertTrue(dateFormat.parse(values[0]).before(dateFormat.parse(values[1])));
150             //next column is mean, ensure it can be parsed
151             Double.parseDouble(values[2]);
152             //last column should be empty, so make sure last char on string is comma
153             assertEquals(',', line.toString().charAt(line.toString().length() - 1));
154         }
155     }

157     protected static class TestLoggingThread extends Thread {
158         protected static final AtomicInteger index = new AtomicInteger();
159 
160         public static final int STOP_WATCH_COUNT = 20;
161 
162         public void run() {
163             String loggingTag = "tag" + index.getAndIncrement();
164 
165             for (int i = 0; i < STOP_WATCH_COUNT; i++) {
166                 StopWatch stopWatch = new Slf4JStopWatch(loggingTag);
167                 int sleepTime = (int) (Math.random() * 400);
168                 try { Thread.sleep(sleepTime); } catch (Exception e) { /*do nothing*/ }
169                 stopWatch.stop();
170             }
171 
172             //this last sleep here seems necessary because otherwise Thread.join() above is returning before
173             //we're really done.
174             try { Thread.sleep(10); } catch (Exception e) { /*ignore*/ }
175         }
176     }
