16  package net.jperf.helpers;
17  
18  import net.jperf.GroupedTimingStatistics;
19  
20  import junit.framework.TestCase;
21  
22  public class GenericAsyncCoalescingStatisticsAppenderTest extends TestCase {
23  
24  	/**
25  	 * Test implementation of GroupedTimingStatisticsHandler which blocks handle() for 100s. This
26  	 * allows testing of thread interrupt.
27  	 */
28      private static class TestGroupedTimingStatisticsHandler implements GenericAsyncCoalescingStatisticsAppender.GroupedTimingStatisticsHandler {
29  
30          private volatile boolean wasInterrupted = false;
31  
32          public void handle(GroupedTimingStatistics statistics) {
33              try {
34                  // simulate very slow, blocking handler, easy to interrupt
35                  Thread.sleep(100000L);
36              } catch (InterruptedException e) {
37                  System.err.println("Interrupted");
38                  wasInterrupted = true;
39              }
40          }
41  
42          public void error(String errorMessage) {
43              System.err.println("Error logged");
44          }
45      }
46  
47      /**
48       * Tests that the thread calling handler handle() (the Dispatcher thread) is interrupted when
49       * logging-shutdown timeout has expired.
50       */
51      public void testInterruptsThread() throws InterruptedException {
52  
53          GenericAsyncCoalescingStatisticsAppender appender = new GenericAsyncCoalescingStatisticsAppender();
54          TestGroupedTimingStatisticsHandler handler = new TestGroupedTimingStatisticsHandler();
55  
56          // very short timeslice, 1ms
57          appender.setTimeSlice(1L);
58          // very short shutdown wait, 1ms
59          appender.setShutdownWaitMillis(1L);
60  
61          // start the handler and log a single message
62          appender.start(handler);
63          appender.append("start[1230068856846] time[2] tag[tag1]");
64  
65          // sleep to make sure handler thread started and a logging event queued (blocked)
66          Thread.sleep(100L);
67          appender.stop();
68          // sleep to make sure interrupt has been processed
69          Thread.sleep(100L);
70  
71          assertTrue("Handler was not interrupted", handler.wasInterrupted);
72      }
73  }
