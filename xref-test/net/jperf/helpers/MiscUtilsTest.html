17  package net.jperf.helpers;
18  
19  import junit.framework.Assert;
20  import junit.framework.TestCase;
21  
22  import java.util.GregorianCalendar;
23  
24  /**
25   * Tests the helper methods of the MiscUtils class
26   */
27  public class MiscUtilsTest extends TestCase {
28      public void testEscapeStringForCsv() throws Throwable {
29          Assert.assertEquals("\"foo\"", MiscUtils.escapeStringForCsv("foo", new StringBuilder()).toString());
30          Assert.assertEquals("\"foo,bar\"", MiscUtils.escapeStringForCsv("foo,bar", new StringBuilder()).toString());
31          Assert.assertEquals("\"\"\"foo\"\"\"", MiscUtils.escapeStringForCsv("\"foo\"", new StringBuilder()).toString());
32          Assert.assertEquals("\"foo,\"\"bar\"\"\"", MiscUtils.escapeStringForCsv("foo,\"bar\"", new StringBuilder()).toString());
33      }
34  
35      public void testPadIntToTwoDigits() throws Throwable {
36          Assert.assertEquals("00", MiscUtils.padIntToTwoDigits(0, new StringBuilder()).toString());
37          Assert.assertEquals("01", MiscUtils.padIntToTwoDigits(1, new StringBuilder()).toString());
38          Assert.assertEquals("09", MiscUtils.padIntToTwoDigits(9, new StringBuilder()).toString());
39          Assert.assertEquals("10", MiscUtils.padIntToTwoDigits(10, new StringBuilder()).toString());
40          Assert.assertEquals("99", MiscUtils.padIntToTwoDigits(99, new StringBuilder()).toString());
41      }
42  
43      public void testFormatDateIso8601() throws Throwable {
44          Assert.assertEquals("2008-01-01 13:01:10",
45                  MiscUtils.formatDateIso8601(new GregorianCalendar(2008, 0, 1, 13, 1, 10).getTimeInMillis()));
46          Assert.assertEquals("2010-10-31 08:59:59",
47                  MiscUtils.formatDateIso8601(new GregorianCalendar(2010, 9, 31, 8, 59, 59).getTimeInMillis()));
48      }
49  }
