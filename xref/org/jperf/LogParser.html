28  /**
29   * LogParser provides the main method for reading a log of StopWatch output and generating statistics and graphs
30   * from that output. Run "java -jar pathToPerf4jJar --help" for instructions.
31   *
32   * @author Alex Devine
33   */
34  public class LogParser {
35      /**
36       * The input log that is being parsed.
37       */
38      private Reader inputLog;
39      /**
40       * The stream where the GroupedTimingStatistics data will be printed - if null, no statistics will be printed
41       */
42      private PrintStream statisticsOutput;
43      /**
44       * The stream where graphing output will be printed - if null, no graphing output will be printed. May be the
45       * same stream as statisticsOutput.
46       */
47      private PrintStream graphingOutput;
48      /**
49       * The chart generator used to send the mean time graph to graphingOutput.
50       */
51      private StatisticsChartGenerator meanTimeChartGenerator;
52      /**
53       * The chart generator used to send the TPS graph to graphingOutput.
54       */
55      private StatisticsChartGenerator tpsChartGenerator;
56      /**
57       * The length of time, in milliseconds, of the timeslice of each GroupedTimingStatistics.
58       */
59      private long timeSlice;
60      /**
61       * Whether or not "rollup statistics" should be created for each GroupedTimingStatistics created.
62       */
63      private boolean createRollupStatistics;
64      /**
65       * The formatter to use to print statistics.
66       */
67      private StatisticsFormatter statisticsFormatter;
68  
69      // --- Constructors ---
70      /**
71       * Default constructor reads input from standard in, writes statistics output to standard out, does not write
72       * graph output, has a time slice window of 30 seconds, and does not create rollup statistics.
73       */
74      public LogParser() {
75          this(new InputStreamReader(System.in),
76               System.out,
77               null /* no graph output */,
78               30000L,
79               false /* don't create rollup statistics */,
80               new DefaultStatisticsFormatter(new GroupedTimingStatisticsTextFormatter()));
81      }
82  
83      /**
84       * Creates a new LogParser to parse log data from the input.
85       *
86       * @param inputLog               The log being parsed, which should contain {@link org.perf4j.StopWatch} log messages.
87       * @param statisticsOutput       The stream where calculated statistics information should be written - if null,
88       *                               statistics data is not written.
89       * @param graphingOutput         The stream where graphing data should be written - if null, graphs are not written.
90       * @param timeSlice              The length of time, in milliseconds, of the timeslice of each statistics data created.
91       * @param createRollupStatistics Whether or not "rollup statistics" should be created for each timeslice of data.
92       * @param statisticsFormatter    The formatter to use to print GroupedTimingStatistics
93       */
94      public LogParser(Reader inputLog, PrintStream statisticsOutput, PrintStream graphingOutput,
95                       long timeSlice, boolean createRollupStatistics,
96                       StatisticsFormatter statisticsFormatter) {
97          this.inputLog = inputLog;
98          this.statisticsOutput = statisticsOutput;
99          this.graphingOutput = graphingOutput;
100         this.timeSlice = timeSlice;
101         this.createRollupStatistics = createRollupStatistics;
102         if (graphingOutput != null) {
103             this.meanTimeChartGenerator = newMeanTimeChartGenerator();
104             this.tpsChartGenerator = newTpsChartGenerator();
105         }
106         this.statisticsFormatter = statisticsFormatter;
107     }

111     /**
112      * Reads all the data from the inputLog, parses it, and writes the statistics data and graphing data as desired
113      * to the output streams.
114      */
115     public void parseLog() {
116     	if (statisticsOutput != null) {
117             statisticsOutput.print(statisticsFormatter.header());
118         }
119 
120         Iterator<StopWatch> stopWatchIter = new StopWatchLogIterator(inputLog);
121 
122         int i = 0;
123         for (GroupingStatisticsIterator statsIter = new GroupingStatisticsIterator(stopWatchIter,
124                                                                                    timeSlice,
125                                                                                    createRollupStatistics);
126              statsIter.hasNext();) {
127             GroupedTimingStatistics statistics = statsIter.next();
128 
129             if (statisticsOutput != null) {
130                 statisticsOutput.print(statisticsFormatter.format(statistics));
131             }
132 
133             if (graphingOutput != null) {
134                 meanTimeChartGenerator.appendData(statistics);
135                 tpsChartGenerator.appendData(statistics);
136                 if ((++i % StatisticsChartGenerator.DEFAULT_MAX_DATA_POINTS == 0) ||
137                     (!statsIter.hasNext())) {
138                     printGraphOutput();
139                 }
140             }
141         }
142         
143         if (statisticsOutput != null) {
144             statisticsOutput.print(statisticsFormatter.footer());
145         }
146     }
147 
148     protected StatisticsChartGenerator newMeanTimeChartGenerator() {
149         return new GoogleChartGenerator();
150     }
151 
152     protected StatisticsChartGenerator newTpsChartGenerator() {
153         return new GoogleChartGenerator(StatsValueRetriever.TPS_VALUE_RETRIEVER);
154     }
155 
156     protected void printGraphOutput() {
157         graphingOutput.println("<br/><br/><img src=\"" + meanTimeChartGenerator.getChartUrl() + "\"/>");
158         graphingOutput.println("<br/><br/><img src=\"" + tpsChartGenerator.getChartUrl() + "\"/>");
159     }

163     public static void main(String[] args) {
164         System.exit(runMain(args));

167     public static int runMain(String[] args) {
168         try {
169             List<String> argsList = new ArrayList<String>(Arrays.asList(args));
170 
171             if (printUsage(argsList)) {
172                 return 0;
173             }
174             PrintStream statisticsOutput = null;
175         	PrintStream graphingOutput = null;
176         	Reader input = null;
177         	try {
178 	            long timeSlice = getTimeSlice(argsList);
179 	            boolean rollupStatistics = getRollupStatistics(argsList);
180 	            StatisticsFormatter formatter = getStatisticsFormatter(argsList);
181 	            statisticsOutput = openStatisticsOutput(argsList);
182 	            graphingOutput = openGraphingOutput(argsList);
183 	            input = openInput(argsList);
184 	
185 	            if (!argsList.isEmpty()) {
186 	                printUnknownArgs(argsList);
187 	                return 1;
188 	            }
189 	
190 	            new LogParser(input, statisticsOutput, graphingOutput, timeSlice, rollupStatistics, formatter).parseLog();
191         	} finally {
192         		closeInput(input);
193         		closeStatisticsOutput(statisticsOutput);
194         		closeGraphingOutput(graphingOutput);
195         	}
196         } catch ( Exception e ) {
197             e.printStackTrace();
198             return 1;
199         }
200         return 0;
201     }

203     protected static boolean printUsage(List<String> argsList) {
204         if (getIndexOfArg(argsList, false, "-h", "--help", "-?", "--usage") >= 0) {
205             System.out.println("Usage: LogParser [-o|--out|--output outputFile] " +
206                                "[-g|--graph graphingOutputFile] " +
207                                "[-t|--timeslice timeslice] " +
208                                "[-r] " +
209                                "[-f|--format text|csv] " +
210                                "[logInputFile]");
211             System.out.println("Arguments:");
212             System.out.println("  logInputFile - The log file to be parsed. If not specified, log data is read from stdin.");
213             System.out.println("  -o|--out|--output outputFile - The file where generated statistics should be written." +
214                                " If not specified, statistics are written to stdout.");
215             System.out.println("  -g|--graph graphingOutputFile - The file where generated perf graphs should be written." +
216                                " If not specified, no graphs are written.");
217             System.out.println("  -t|--timeslice timeslice - The length of time (in ms) of each timeslice for which" +
218                                " statistics should be generated. Defaults to 30000 ms.");
219             System.out.println("  -r - Whether or not statistics rollups should be generated." +
220                                " If not specified, rollups are not generated.");
221             System.out.println("  -f|--format text|csv - The format for the statistics output, either plain text or CSV." +
222                                " Defaults to text.");
223             System.out.println("                         If format is csv, then the columns output are tag, start, stop, mean, min, max, stddev, and count.");
224             System.out.println();
225             System.out.println("Note that out, stdout, err and stderr can be used as aliases to the standard output" +
226                                " streams when specifying output files.");
227             return true;
228         }
229 
230         return false;
231     }

233     protected static PrintStream openStatisticsOutput(List<String> argsList) throws IOException {
234         int indexOfOut = getIndexOfArg(argsList, true, "-o", "--output", "--out");
235         if (indexOfOut >= 0) {
236             String fileName = argsList.remove(indexOfOut + 1);
237             argsList.remove(indexOfOut);
238             return openStream(fileName);
239         } else {
240             return System.out;
241         }
242     }
243 
244     protected static PrintStream openGraphingOutput(List<String> argsList) throws IOException {
245         int indexOfOut = getIndexOfArg(argsList, true, "-g", "--graph");
246         if (indexOfOut >= 0) {
247             String fileName = argsList.remove(indexOfOut + 1);
248             argsList.remove(indexOfOut);
249             PrintStream retVal = openStream(fileName);
250             retVal.println("<html>");
251             retVal.println("<head><title>Perf4J Performance Graphs</title></head>");
252             retVal.println("<body>");
253             return retVal;
254         } else {
255             return null;
256         }
257     }
258     
259     protected static void closeInput(Reader input) throws IOException {
260         if (input != null) {
261         	// InputStreamReader is used for stdin
262             if (!(input instanceof InputStreamReader)) {
263             	input.close();
264             }
265         }
266     }
267     
268     protected static void closeStatisticsOutput(PrintStream statisticsOutput) throws IOException {
269         if (statisticsOutput != null) {
270             if (statisticsOutput != System.out && statisticsOutput != System.err) {
271             	statisticsOutput.close();
272             }
273         }
274     }
275 
276     protected static void closeGraphingOutput(PrintStream graphingOutput) throws IOException {
277         if (graphingOutput != null) {
278             graphingOutput.println("</body></html>");
279             if (graphingOutput != System.out && graphingOutput != System.err) {
280                 graphingOutput.close();
281             }
282         }
283     }

285     protected static long getTimeSlice(List<String> argsList) {
286         int indexOfOut = getIndexOfArg(argsList, true, "-t", "--timeslice");
287         if (indexOfOut >= 0) {
288             String timeslice = argsList.remove(indexOfOut + 1);
289             argsList.remove(indexOfOut);
290             return Long.parseLong(timeslice);
291         } else {
292             return 30000L;
293         }
294     }
295 
296     protected static boolean getRollupStatistics(List<String> argsList) {
297         int indexOfOut = getIndexOfArg(argsList, false, "-r", "--rollup");
298         if (indexOfOut >= 0) {
299             argsList.remove(indexOfOut);
300             return true;
301         } else {
302             return false;
303         }
304     }

306     protected static StatisticsFormatter getStatisticsFormatter(List<String> argsList) {
307         int indexOfFormat = getIndexOfArg(argsList, true, "-f", "--format");
308         if (indexOfFormat >= 0) {
309             String formatString = argsList.remove(indexOfFormat + 1);
310             argsList.remove(indexOfFormat);
311             if ("text".equalsIgnoreCase(formatString)) {
312                 return new DefaultStatisticsFormatter(new GroupedTimingStatisticsTextFormatter());
313             } else if ("csv".equalsIgnoreCase(formatString)) {
314                 return new DefaultStatisticsFormatter(new GroupedTimingStatisticsCsvFormatter());
315             } else if (formatString.startsWith("json")) {
316                 return new GroupedTimingStatisticsJsonFormatter(formatString);
317             } else {
318                 throw new IllegalArgumentException("Unknown format type: " + formatString);
319             }
320         } else {
321             return new DefaultStatisticsFormatter(new GroupedTimingStatisticsTextFormatter());
322         }
323     }

325     protected static Reader openInput(List<String> argsList) throws IOException {
326         if (argsList.isEmpty()) {
327             return new InputStreamReader(System.in);
328         } else {
329             String fileName = argsList.remove(0);
330             return new BufferedReader(new FileReader(fileName));
331         }
332     }

334     protected static void printUnknownArgs(List<String> argsList) {
335         System.out.println("Unknown arguments: ");
336         for (String arg : argsList) {
337             System.out.print(arg + " ");
338         }
339         System.out.println();
340     }

342     protected static int getIndexOfArg(List<String> args, boolean needsParam, String... argNames) {
343         int retVal = -1;
344         boolean foundArg = false;
345         for (String argName : argNames) {
346             int argIndex = args.indexOf(argName);
347             if (argIndex >= 0) {
348                 if (foundArg) {
349                     throw new IllegalArgumentException("You must specify only one of " + Arrays.toString(argNames));
350                 }
351                 retVal = argIndex;
352                 foundArg = true;
353             }
354         }
355 
356         if ((retVal >= 0) && needsParam && (retVal == args.size() - 1)) {
357             throw new IllegalArgumentException("You must specify a parameter for the " + args.get(retVal) + " arg");
358         }
359 
360         return retVal;
361     }
362 
363     protected static PrintStream openStream(String fileName) throws IOException {
364         if ("stdout".equals(fileName) || "out".equals(fileName)) {
365             return System.out;
366         } else if ("stderr".equals(fileName) || "err".equals(fileName)) {
367             return System.err;
368         } else {
369             return new PrintStream(new FileOutputStream(fileName), true);
370         }
371     }
