360     /**
361      * This method returns true if the Logger it uses is enabled at the normalPriority level of this StopWatch.
362      *
363      * @return true if this StopWatch will output log messages when one of the stop or lap messages that does NOT
364      *         take an exception is called.
365      */
366     public boolean isLogging() {
367         return isLogging(normalPriority);
368     }

370     /**
371      * The log message is overridden to use the Apache Commons Logging Logger to persist the stop watch.
372      *
373      * @param stopWatchAsString The stringified view of the stop watch for logging.
374      * @param exception         An exception, if any, that was passed to the stop or lap method. If this is null then
375      *                          logging will occur at normalPriority, if non-null it will occur at exceptionPriority.
376      */
377     protected void log(String stopWatchAsString, Throwable exception) {
378         log(stopWatchAsString, exception, (exception == null) ? normalPriority : exceptionPriority);
379     }
380 
381     /**
382      * Since Commons Logging doesn't have a first class notion of Level objects or integer values, this method
383      * converts the level value to one of the <tt>isXYZEnabled()</tt> methods on the Log.
384      *
385      * @param atLevel The level at which a log message is logged.
386      * @return true if the Logger used by this StopWatch will output messages at the level specified.
387      */
388     protected boolean isLogging(int atLevel) {
389         switch (atLevel) {
390         case TRACE_LEVEL:
391             return logger.isTraceEnabled();
392         case DEBUG_LEVEL:
393             return logger.isDebugEnabled();
394         case INFO_LEVEL:
395             return logger.isInfoEnabled();
396         case WARN_LEVEL:
397             return logger.isWarnEnabled();
398         case ERROR_LEVEL:
399             return logger.isErrorEnabled();
400         default:
401             // if here it means the level was set to a non-standard value.
402             // Only expend the time to find the closest known level if we need to.
403             return isLogging(closestKnownLevel(atLevel));
404         }
405     }
406 
407     /**
408      * Since Commons Logging doesn't have a first class notion of Level objects or integer values, this method
409      * maps the atLevel parameter to one of the <tt>trace()</tt>, <tt>debug()</tt>, <tt>info()</tt>, <tt>warn()</tt>,
410      * <tt>error()</tt>, or <tt>fatal()</tt> methods on the Log.
411      *
412      * @param stopWatchAsString The stringified view of the stop watch for logging.
413      * @param exception         An exception, if any, that was passed to the stop or lap method. If this is null then
414      *                          logging will occur at normalPriority, if non-null it will occur at exceptionPriority.
415      * @param atLevel           The level at which logging should occur.
416      */
417     protected void log(String stopWatchAsString, Throwable exception, int atLevel) {
418         switch (atLevel) {
419         case TRACE_LEVEL:
420             logger.trace(stopWatchAsString, exception);
421             break;
422         case DEBUG_LEVEL:
423             logger.debug(stopWatchAsString, exception);
424             break;
425         case INFO_LEVEL:
426             logger.info(stopWatchAsString, exception);
427             break;
428         case WARN_LEVEL:
429             logger.warn(stopWatchAsString, exception);
430             break;
431         case ERROR_LEVEL:
432             logger.error(stopWatchAsString, exception);
433             break;
434         default:
435             // if here it means the level was set to a non-standard value.
436             // Only expend the time to find the closest known level if we need to.
437             log(stopWatchAsString, exception, closestKnownLevel(atLevel));
438         }
439     }
440 
441     private int closestKnownLevel(int level) {
442         if (level <= TRACE_LEVEL) {
443             return TRACE_LEVEL;
444         }
445         if (level <= DEBUG_LEVEL) {
446             return DEBUG_LEVEL;
447         }
448         if (level <= INFO_LEVEL) {
449             return INFO_LEVEL;
450         }
451         if (level <= WARN_LEVEL) {
452             return WARN_LEVEL;
453         }
454         return ERROR_LEVEL;
455     }

457     // --- Static Utility Methods ---
458     /**
459      * This utility method provides the standard mapping between log4j level names (which Perf4J uses as the standard
460      * set of possible levels) to the corresponding ..._LEVEL constant from this class.
461      *
462      * @param levelName The name of the logging level, should be one of TRACE, DEBUG, INFO, WARN, ERROR or FATAL.
463      * @return The corresponding ..._LEVEL constant from this class.
464      */
465     public static int mapLevelName(String levelName) {
466         levelName = levelName.toUpperCase();
467         if ("TRACE".equals(levelName)) { return TRACE_LEVEL; }
468         if ("DEBUG".equals(levelName)) { return DEBUG_LEVEL; }
469         if ("INFO".equals(levelName)) { return INFO_LEVEL; }
470         if ("WARN".equals(levelName)) { return WARN_LEVEL; }
471         if ("ERROR".equals(levelName)) { return ERROR_LEVEL; }
472         if ("FATAL".equals(levelName)) { return ERROR_LEVEL; } //SLF4J has no FATAL level
473         return INFO_LEVEL;
474     }

482     private void writeObject(ObjectOutputStream stream) throws IOException {
