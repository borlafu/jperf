1   package net.jperf.slf4j.aop;

3   import net.jperf.aop.AbstractEjbTimingAspect;
4   import net.jperf.slf4j.Slf4JStopWatch;
5   import org.slf4j.LoggerFactory;
6   
7   /**
8    * This TimingAspect implementation uses a SLF4J Logger instance to persist StopWatch log messages.
9    * To use this interceptor in your code, you should add this class name to the {@link javax.interceptor.Interceptors}
10   * annotation on the EJB to be profiled.
11   *
12   * @author Alex Devine
13   */
14  public class EjbTimingAspect extends AbstractEjbTimingAspect {
15      protected Slf4JStopWatch newStopWatch(String loggerName, String levelName) {
16          int levelInt = Slf4JStopWatch.mapLevelName(levelName);
17          return new Slf4JStopWatch(LoggerFactory.getLogger(loggerName), levelInt, levelInt);
18      }
19  }
