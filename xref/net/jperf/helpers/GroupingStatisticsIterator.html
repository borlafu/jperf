16  package net.jperf.helpers;
17  
18  import net.jperf.GroupedTimingStatistics;
19  import net.jperf.StopWatch;
20  
21  import java.util.Iterator;
22  import java.util.NoSuchElementException;
23  
24  /**
25   * This Iterator wraps a {@link StopWatchLogIterator} to return a single {@link GroupedTimingStatistics} object for
26   * each time slice detected. Thus, this iterator is a "gearing" iterator - if there are on average 100 StopWatches
27   * logged during each time slice, the underlying StopWatchLogIterator will return 100 StopWatches for each single
28   * GroupedTimingStatistics object returned by this Iterator.
29   * <p/>
30   * Note that it's assumed that the StopWatch Iterator is ordered according to start time. If this is not true, then
31   * this class will create GroupedTimingStatistics that may reflect StopWatch data from a previous time slice.
32   *
33   * @author Alex Devine
34   */
35  public class
36          GroupingStatisticsIterator implements Iterator<GroupedTimingStatistics> {
37      /**
38       * The underlying StopWatch iterator
39       */
40      private Iterator<StopWatch> stopWatchIterator;
41      /**
42       * The length of each time slice, in milliseconds.
43       */
44      private long timeSlice;
45      /**
46       * Whether or not entries for "rollup" tags should be created in each GroupedTimingStatistics returned.
47       */
48      private boolean createRollupStatistics;
49  
50      /**
51       * This hasNext is really a tri-state var - null indicates I don't know if there's a next one or not.
52       */
53      private Boolean hasNext = null;
54      /**
55       * The next GroupedTimingStatistics to be returned.
56       */
57      private GroupedTimingStatistics nextGroupedTimingStatistics = null;
58      /**
59       * Keeps track of the CURRENT GroupedTimingStatistics while we iterate over the underlying StopWatches
60       */
61      private GroupedTimingStatistics currentGroupedTimingStatistics = new GroupedTimingStatistics();
62      /**
63       * The end time, in milliseconds since the epoch, of the next time slice.
64       */
65      private long nextTimeSliceEndTime = 0L;
66  
67      /**
68       * Creates a GroupingStatisticsIterator that groups StopWatch instances pulled from the specified
69       * stopWatchIterator into GroupedTimingStatistics. A timeslice of 30 seconds is used and rollup statistics are not
70       * created.
71       *
72       * @param stopWatchIterator The StopWatch Iterator that provides the StopWatch instances. If stopWatchIterator
73       *                          returns a null value, will check to see if a timeslice is over and return
74       *                          GroupedTimingStatistics if necessary.
75       */
76      public GroupingStatisticsIterator(Iterator<StopWatch> stopWatchIterator) {
77          this(stopWatchIterator, 30000L, false);
78      }
79  
80      /**
81       * Creates a GroupingStatisticsIterator that groups StopWatch instances pulled from the specified
82       * stopWatchIterator into GroupedTimingStatistics.
83       *
84       * @param stopWatchIterator      The StopWatch Iterator that provides the StopWatch instances. If stopWatchIterator
85       * 								 returns a null value, will check to see if a timeslice is over and return
86       * 								 GroupedTimingStatistics if necessary.
87       * @param timeSlice              The length of each time slice, in milliseconds.
88       * @param createRollupStatistics Whether or not entries for "rollup" tags should be created
89       */
90      public GroupingStatisticsIterator(Iterator<StopWatch> stopWatchIterator,
91                                        long timeSlice,
92                                        boolean createRollupStatistics) {
93          this.stopWatchIterator = stopWatchIterator;
94          this.timeSlice = timeSlice;
95          this.createRollupStatistics = createRollupStatistics;
96          this.currentGroupedTimingStatistics.setCreateRollupStatistics(createRollupStatistics);
97      }
98  
99      public boolean hasNext() {
100         //if I don't know the state of next, pull the next statistics to determine the state of next
101         if (hasNext == null) {
102             nextGroupedTimingStatistics = getNext();
103             hasNext = (nextGroupedTimingStatistics != null);
104         }
105         return hasNext;
106     }
107 
108     public GroupedTimingStatistics next() {
109         //if I already determined I don't have a next, throw an exception
110         if (Boolean.FALSE.equals(hasNext)) {
111             throw new NoSuchElementException();
112         }
113 
114         //if I don't know what to return yet, find out - note this only happens if I call next() before a call
115         //to hasNext().
116         if (nextGroupedTimingStatistics == null) {
117             nextGroupedTimingStatistics = getNext();
118 
119             //if there's still nothing I'm done
120             if (nextGroupedTimingStatistics == null) {
121                 hasNext = false;
122                 throw new NoSuchElementException();
123             }
124         }
125 
126         //before I return, clear the state of the variables used to determine the next value.
127         GroupedTimingStatistics retVal = nextGroupedTimingStatistics;
128         hasNext = null;
129         nextGroupedTimingStatistics = null;
130         return retVal;
131     }
132 
133     /**
134      * Remove is not supported.
135      *
136      * @throws UnsupportedOperationException Always thrown.
137      */
138     public void remove() {
139         throw new UnsupportedOperationException();
140     }
141 
142     /**
143      * Helper method runs over the StopWatch Iterator to group the StopWatches into GroupedTimingStatistics.
144      *
145      * @return The next GroupedTimingStatistics from the underlying StopWatch Iterator, or null if there are no
146      *         StopWatch instances left.
147      */
148     private GroupedTimingStatistics getNext() {
149         while (stopWatchIterator.hasNext()) {
150             StopWatch stopWatch = stopWatchIterator.next();
151             
152             // if stopwatch is null, then the timeslice might be over (use current time)
153             long startTime = stopWatch == null ? System.currentTimeMillis() : stopWatch.getStartTime();
154             
155             //the first time we pull a stop watch we need to set the first end time
156             if (nextTimeSliceEndTime == 0L) {
157                 nextTimeSliceEndTime = ((startTime / timeSlice) * timeSlice) + timeSlice;
158             }
159 
160             if (startTime >= nextTimeSliceEndTime) {
161                 //then we're over a new time boundary, so update the current timing statistics and return it.
162                 currentGroupedTimingStatistics.setStartTime(nextTimeSliceEndTime - timeSlice);
163                 currentGroupedTimingStatistics.setStopTime(nextTimeSliceEndTime);
164                 GroupedTimingStatistics retVal = currentGroupedTimingStatistics;
165 
166                 //set the state for the next slice
167                 currentGroupedTimingStatistics = new GroupedTimingStatistics();
168                 currentGroupedTimingStatistics.setCreateRollupStatistics(createRollupStatistics);
169                 if (stopWatch != null) {
170                 	// only add if we got a new stopwatch, not if timeslice just expired
171                 	currentGroupedTimingStatistics.addStopWatch(stopWatch);
172                 }                
173                 nextTimeSliceEndTime = ((startTime / timeSlice) * timeSlice) + timeSlice;
174                 return retVal;
175             } else if (stopWatch != null) {
176                 currentGroupedTimingStatistics.addStopWatch(stopWatch);
177             }
178         }
179 
180         //if here then there are no more stopwatches left, so clean up the last batch
181         if (!currentGroupedTimingStatistics.getStatisticsByTag().isEmpty()) {
182             currentGroupedTimingStatistics.setStartTime(nextTimeSliceEndTime - timeSlice);
183             currentGroupedTimingStatistics.setStopTime(nextTimeSliceEndTime);
184             GroupedTimingStatistics retVal = currentGroupedTimingStatistics;
185 
186             //create an empty GroupedTimingStatistics so we know to return null in the next call to this method.
187             currentGroupedTimingStatistics = new GroupedTimingStatistics();
188             currentGroupedTimingStatistics.setCreateRollupStatistics(createRollupStatistics);
189 
190             return retVal;
191         } else {
192             //The StopWatch iterator is done and we already printed the last GroupedTimingStatistics batch
193             return null;
194         }
195     }
196 }
