16  package net.jperf.helpers;
17  
18  import net.jperf.TimingStatistics;
19  
20  import java.util.Map;
21  import java.util.Collections;
22  import java.util.LinkedHashMap;
23  
24  /**
25   * The StatsValueRetriever is used to enable retrieval of any of the statistics on the TimingStatistics object
26   * by name. In addition, retrieval of a transactions per second statistic is supported.
27   *
28   * @author Alex Devine
29   */
30  public abstract class StatsValueRetriever {
31      public static final StatsValueRetriever MEAN_VALUE_RETRIEVER = new StatsValueRetriever() {
32          public Number getStatsValue(TimingStatistics timingStats, long windowLength) {
33              return (timingStats == null) ? 0.0 : timingStats.getMean();
34          }
35  
36          public Class<Double> getValueClass() { return Double.class; }
37  
38          public String getValueName() { return "Mean"; }
39      };
40  
41      public static final StatsValueRetriever STD_DEV_VALUE_RETRIEVER = new StatsValueRetriever() {
42          public Number getStatsValue(TimingStatistics timingStats, long windowLength) {
43              return (timingStats == null) ? 0.0 : timingStats.getStandardDeviation();
44          }
45  
46          public Class<Double> getValueClass() { return Double.class; }
47  
48          public String getValueName() { return "StdDev"; }
49      };
50  
51      public static final StatsValueRetriever MIN_VALUE_RETRIEVER = new StatsValueRetriever() {
52          public Number getStatsValue(TimingStatistics timingStats, long windowLength) {
53              return (timingStats == null) ? 0L : timingStats.getMin();
54          }
55  
56          public Class<Long> getValueClass() { return Long.class; }
57  
58          public String getValueName() { return "Min"; }
59      };
60  
61      public static final StatsValueRetriever MAX_VALUE_RETRIEVER = new StatsValueRetriever() {
62          public Number getStatsValue(TimingStatistics timingStats, long windowLength) {
63              return (timingStats == null) ? 0L : timingStats.getMax();
64          }
65  
66          public Class<Long> getValueClass() { return Long.class; }
67  
68          public String getValueName() { return "Max"; }
69      };
70  
71      public static final StatsValueRetriever COUNT_VALUE_RETRIEVER = new StatsValueRetriever() {
72          public Number getStatsValue(TimingStatistics timingStats, long windowLength) {
73              return (timingStats == null) ? 0 : timingStats.getCount();
74          }
75  
76          public Class<Integer> getValueClass() { return Integer.class; }
77  
78          public String getValueName() { return "Count"; }
79      };
80  
81      public static final StatsValueRetriever TPS_VALUE_RETRIEVER = new StatsValueRetriever() {
82          public Number getStatsValue(TimingStatistics timingStats, long windowLength) {
83              return (timingStats == null || windowLength == 0) ?
84                     0.0 :
85                     ((double) timingStats.getCount()) / (((double) windowLength) / 1000.0);
86          }
87  
88          public Class<Double> getValueClass() { return Double.class; }
89  
90          public String getValueName() { return "TPS"; }
91      };
92  
93      /**
94       * Default unmodifiable Map of statistic name to the corresponding StatsValueRetriever object that retrieves that
95       * statistic. Statistic names are Mean, StdDev, Min, Max, Count and TPS.
96       */
97      public static final Map<String, StatsValueRetriever> DEFAULT_RETRIEVERS;
98      static {
99          Map<String, StatsValueRetriever> defaultRetrievers = new LinkedHashMap<String, StatsValueRetriever>();
100         defaultRetrievers.put(MEAN_VALUE_RETRIEVER.getValueName(), MEAN_VALUE_RETRIEVER);
101         defaultRetrievers.put(STD_DEV_VALUE_RETRIEVER.getValueName(), STD_DEV_VALUE_RETRIEVER);
102         defaultRetrievers.put(MIN_VALUE_RETRIEVER.getValueName(), MIN_VALUE_RETRIEVER);
103         defaultRetrievers.put(MAX_VALUE_RETRIEVER.getValueName(), MAX_VALUE_RETRIEVER);
104         defaultRetrievers.put(COUNT_VALUE_RETRIEVER.getValueName(), COUNT_VALUE_RETRIEVER);
105         defaultRetrievers.put(TPS_VALUE_RETRIEVER.getValueName(), TPS_VALUE_RETRIEVER);
106         DEFAULT_RETRIEVERS = Collections.unmodifiableMap(defaultRetrievers);
107     }
108 
109     /**
110      * Retrieves a single statistic value from the specified TimingStatistics object.
111      *
112      * @param timingStats  The TimingStatistics object containing the data to be retrieved.
113      *                     May be null, if so 0 is returned.
114      * @param windowLength The length of time, in milliseconds, of the data window represented by the TimingStatistics.
115      * @return The value requested.
116      */
117     public abstract Number getStatsValue(TimingStatistics timingStats, long windowLength);
118 
119     /**
120      * Gets the class of the object returned by {@link #getStatsValue(net.jperf.TimingStatistics, long)}.
121      *
122      * @return The value class.
123      */
124     public abstract Class<? extends Number> getValueClass();
125 
126     /**
127      * Returns the name of the value, such as "Mean" or "Max".
128      *
129      * @return The name of the value retrieved.
130      */
131     public abstract String getValueName();
132 }
