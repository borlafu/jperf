17  package net.jperf.helpers;
18  
19  import java.io.Serializable;
20  import java.util.regex.Matcher;
21  import java.util.regex.Pattern;
22  
23  /**
24   * Instances of this class are used by the StatisticsExposingMBean to determine if JMX notifications should be sent if
25   * timing statistics fall outside a specified acceptable range.
26   *
27   * @author Alex Devine
28   */
29  public class AcceptableRangeConfiguration implements Serializable, Cloneable {
30      private static final long serialVersionUID = -1386668939617425680L;
31      private String attributeName;
32      private double minValue = Double.NEGATIVE_INFINITY;
33      private double maxValue = Double.POSITIVE_INFINITY;
34  
35      protected static final Pattern CONFIG_STRING_PATTERN = Pattern.compile("(.+?)\\((<(.+?)|>(.+?)|(.+?)-(.+?))\\)");
36  
37      // --- Constructors ---
38  
39      /**
40       * Default constructor allows the attributeName, minValue and maxValue properties to be set later with
41       * the setter methods.
42       */
43      public AcceptableRangeConfiguration() { }
44  
45      /**
46       * Parses a configuration string to get the attributeName, minValue and maxValue properties. The format of the
47       * config string should be one of:
48       * <ul>
49       * <li>attributeName(&lt;maxValue), where maxValue is the maximum possible acceptable value.
50       * <li>attributeName(&gt;minValue), where minValue is the maximum possible acceptable value.
51       * <li>attributeName(minValue-maxValue), to specify a range for acceptable values.
52       * </ul>
53       *
54       * @param configString The configString to parse
55       * @throws IllegalArgumentException Thrown if the configString did not use the acceptable format.
56       */
57      public AcceptableRangeConfiguration(String configString) {
58          Matcher matcher = CONFIG_STRING_PATTERN.matcher(configString);
59          if (matcher.matches()) {
60              attributeName = matcher.group(1).trim();
61  
62              try {
63                  String rangeString = matcher.group(2).trim();
64                  if (rangeString.startsWith("<")) {
65                      maxValue = Double.parseDouble(matcher.group(3).trim());
66                  } else if (rangeString.startsWith(">")) {
67                      minValue = Double.parseDouble(matcher.group(4).trim());
68                  } else {
69                      minValue = Double.parseDouble(matcher.group(5).trim());
70                      maxValue = Double.parseDouble(matcher.group(6).trim());
71                  }
72              } catch (Exception e) {
73                  throw new IllegalArgumentException("Invalid acceptable range config string: " + configString);
74              }
75          } else {
76              throw new IllegalArgumentException("Invalid acceptable range config string: " + configString);
77          }
78      }
79  
80      /**
81       * Creates a new AcceptableRangeConfiguration with the specified attributeName, min and max values.
82       *
83       * @param attributeName The name of the MBean attribute that is being constrained.
84       * @param minValue      The minimum acceptable value
85       * @param maxValue      The maximum acceptable value
86       */
87      public AcceptableRangeConfiguration(String attributeName, double minValue, double maxValue) {
88          this.attributeName = attributeName;
89          this.minValue = minValue;
90          this.maxValue = maxValue;
91      }
92  
93      // --- Bean Methods ---
94  
95      public String getAttributeName() { return attributeName; }
96  
97      public void setAttributeName(String attributeName) { this.attributeName = attributeName; }
98  
99      public double getMinValue() { return minValue; }
100 
101     public void setMinValue(double minValue) { this.minValue = minValue; }
102 
103     public double getMaxValue() { return maxValue; }
104 
105     public void setMaxValue(double maxValue) { this.maxValue = maxValue; }
106 
107     // --- Utility Methods ---
108 
109     /**
110      * Determines whether or not the specified value is within the acceptable range.
111      *
112      * @param value The value to check
113      * @return returns true if value >= minValue && value <= maxValue, false otherwise
114      */
115     public boolean isInRange(double value) {
116         return value >= minValue && value <= maxValue;
117     }
118 
119     // --- Object Methods ---
120 
121     public boolean equals(Object o) {
122         if (this == o) {
123             return true;
124         }
125         if (!(o instanceof AcceptableRangeConfiguration)) {
126             return false;
127         }
128 
129         AcceptableRangeConfiguration that = (AcceptableRangeConfiguration) o;
130 
131         return Double.compare(that.maxValue, maxValue) == 0 && 
132                Double.compare(that.minValue, minValue) == 0 &&
133                (attributeName == null ? that.attributeName == null : attributeName.equals(that.attributeName));
134     }
135 
136     public int hashCode() {
137         int result;
138         long temp;
139         result = (attributeName != null ? attributeName.hashCode() : 0);
140         temp = minValue != +0.0d ? Double.doubleToLongBits(minValue) : 0L;
141         result = 31 * result + (int) (temp ^ (temp >>> 32));
142         temp = maxValue != +0.0d ? Double.doubleToLongBits(maxValue) : 0L;
143         result = 31 * result + (int) (temp ^ (temp >>> 32));
144         return result;
145     }
146 
147     public String toString() {
148         if (minValue == Double.NEGATIVE_INFINITY) {
149             return attributeName + "(<" + maxValue + ")";
150         } else if (maxValue == Double.POSITIVE_INFINITY) {
151             return attributeName + "(>" + minValue + ")";
152         } else {
153             return attributeName + "(" + minValue + "-" + maxValue + ")";
154         }
155     }
156 
157     public AcceptableRangeConfiguration clone() {
158         try {
159             return (AcceptableRangeConfiguration) super.clone();
160         } catch (Exception e) {
161             throw new Error("Unexpected CloneNotSupportedException");
162         }
163     }
164 }
