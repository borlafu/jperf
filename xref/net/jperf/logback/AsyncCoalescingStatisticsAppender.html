33  /**
34   * This logback Appender groups StopWatch log messages together to form GroupedTimingStatistics. At a scheduled interval
35   * the StopWatch log messages that currently exist in the buffer are pulled to create a single
36   * GroupedTimingStatistics instance that is then sent to any attached appenders.
37   * <p/>
38   * Note that any LoggingEvents which do NOT contain StopWatch objects are discarded. Also, this appender stores logged
39   * messages in a bounded buffer before sending those messages to downstream appenders. If the buffer becomes full then
40   * subsequent logs will be discarded until the buffer has time to clear. You can access the number of discarded
41   * messages using the getNumDiscardedMessages() method.
42   *
43   * @author Alex Devine
44   * @author Xu Huisheng
45   */
46  public class AsyncCoalescingStatisticsAppender extends AppenderBase<LoggingEvent> implements AppenderAttachable<LoggingEvent> {
47      // --- configuration options ---
48      // note most configuration options are provided by the GenericAsyncCoalescingStatisticsAppender
49      /**
50       * DownstreamLogLevel option, converted to a Level object
51       */
52      private Level downstreamLogLevel = Level.INFO;
53  
54      // --- contained objects ---
55      /**
56       * This instance provides the main logic for this appender. This wrapper class just provides the logback-specific
57       * parts.
58       */
59      private final GenericAsyncCoalescingStatisticsAppender baseImplementation =
60              newGenericAsyncCoalescingStatisticsAppender();
61  
62      /**
63       * The downstream appenders are contained in this AppenderAttachableImpl
64       */
65      private final AppenderAttachableImpl<LoggingEvent> downstreamAppenders = new AppenderAttachableImpl<LoggingEvent>();

72       *
73       * @return the TimeSlice option.

110     /**
111      * The <b>DownstreamLogLevel</b> option gets the Level of the GroupedTimingStatistics LoggingEvent that is sent to
112      * downstream appenders. Since each GroupedTimingStatistics represents a view of a collection of single StopWatch
113      * timing event, each of which may have been logged at different levels, this appender needs to decide on a single
114      * Level to use to notify downstream appenders. Defaults to "INFO".
115      *
116      * @return The DownstreamLogLevel option as a String
117      */
118     public String getDownstreamLogLevel() {
119         return downstreamLogLevel.toString();
120     }
121 
122     /**
123      * Sets the value of the <b>DownstreamLogLevel</b> option. This String must be one of the defined Level constants.
124      *
125      * @param downstreamLogLevel The new DownstreamLogLevel option.
126      */
127     public void setDownstreamLogLevel(String downstreamLogLevel) {
128         this.downstreamLogLevel = Level.toLevel(downstreamLogLevel);
129     }

131     /**
132      * The <b>CreateRollupStatistics</b> option is used to determine whether "rollup" statistics should be created.
133      * If the tag name of a StopWatch in a log message contains periods, then the GroupedTimingStatistics will be
134      * created as if each substring of the tag up to the period was also logged with a separate StopWatch instance.
135      * For example, suppose a StopWatch was logged with a tag of "requests.specificReq.PASS". For grouping purposes
136      * a StopWatch entry would be logged under each of the following tags:
137      * <ul>
138      * <li>requests
139      * <li>requests.specificReq
140      * <li>requests.specificReq.PASS
141      * </ul>
142      * This allows you to view statistics at both an individual and aggregated level. If there were other StopWatch
143      * entries with a tag of "requests.specificReq.FAIL", then the data collected at the "requests.specificReq" level
144      * would include BOTH PASS and FAIL events.
145      *
146      * @return The CreateRollupStatistics option.
147      */
148     public boolean isCreateRollupStatistics() {
149         return baseImplementation.isCreateRollupStatistics();
150     }
151

153      * Sets the value of the <b>CreateRollupStatistics</b> option.

159     }

206     public void start() {
207         super.start();
208         //Start the underlying generic appender with a handler object that pumps statistics to the downstream appenders
209         baseImplementation.start(new GenericAsyncCoalescingStatisticsAppender.GroupedTimingStatisticsHandler() {
210             public void handle(GroupedTimingStatistics statistics) {
211                 LoggingEvent coalescedLoggingEvent =
212                         new LoggingEvent(Logger.class.getName(),
213                                          getLoggerContext().getLogger(StopWatch.DEFAULT_LOGGER_NAME),
214                                          downstreamLogLevel,
215                                          "{}",
216                                          null,
217                                          new Object[] {statistics});
218                 try {
219                     synchronized(downstreamAppenders) {
220                         downstreamAppenders.appendLoopOnAppenders(coalescedLoggingEvent);
221                     }
222                 } catch (Exception e) {
223                     addError("Exception calling append with GroupedTimingStatistics on downstream appender",
224                             e);
225                 }
226             }
227 
228             public void error(String errorMessage) {
229                 addError(errorMessage);
230             }
231         });
232 
233     }

288     // --- appender methods ---
289     protected void append(LoggingEvent event) {
290         baseImplementation.append(event.getFormattedMessage());
291     }
292 
293     public void stop() {
294         baseImplementation.stop();
295 
296         //close the downstream appenders
297         synchronized (downstreamAppenders) {
298             //first FLUSH any flushable downstream appenders (fix for PERFFORJ-22). Note we CAN NOT just flush and
299             //close in one loop because this breaks in the case of a "diamond" relationship between appenders, where,
300             //say, this appender has 2 attached GraphingStatisticsAppenders that each write to a SINGLE attached
301             //FileAppender.
302             for (Iterator<Appender<LoggingEvent>> iter = downstreamAppenders.iteratorForAppenders();
303                  iter != null && iter.hasNext();) {
304                 Appender<LoggingEvent> appender = iter.next();
305                 if (appender instanceof Flushable) {
306                     try {
307                         ((Flushable)appender).flush();
308                     } catch (Exception e) { /* Just eat the exception, we're closing down */ }
309                 }
310             }
311 
312             //THEN close them
313             for (Iterator<Appender<LoggingEvent>> iter = downstreamAppenders.iteratorForAppenders();
314                 iter != null && iter.hasNext();) {
315                 iter.next().stop();
316             }
317         }
318         super.stop();
319     }
