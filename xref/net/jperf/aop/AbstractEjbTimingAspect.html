18  package net.jperf.aop;
19  
20  import net.jperf.LoggingStopWatch;
21  
22  import javax.interceptor.AroundInvoke;
23  import javax.interceptor.InvocationContext;
24  import java.lang.reflect.Method;
25  
26  /**
27   * This is the base class for TimingAspects that use the EJB interceptor framework.
28   * Subclasses just need to implement the {@link #newStopWatch} method to use their logging framework of choice
29   * (e.g. log4j or java.logging) to persist the StopWatch log message.
30   *
31   * @author Alex Devine
32   */
33  public abstract class AbstractEjbTimingAspect extends AgnosticTimingAspect {
34      /**
35       * This is the interceptor that runs the target method, surrounding it with stop watch start and stop calls.
36       *
37       * @param ctx The InvocationContext will be passed in by the Java EE server.
38       * @return The return value from the executed method.
39       * @throws Exception Any exceptions thrown by the executed method will bubble up.
40       */
41      @AroundInvoke
42      public Object doPerfLogging(final InvocationContext ctx) throws Exception {
43          final Method executingMethod = ctx.getMethod();
44  
45          //need to get the Profiled annotation off the method, otherwise use a default
46          Profiled profiled = (executingMethod == null) ?
47                              DefaultProfiled.INSTANCE :
48                              ctx.getMethod().getAnnotation(Profiled.class);
49          if (profiled == null) {
50              profiled = DefaultProfiled.INSTANCE;
51          }
52  
53          //note - the EJB 3.0 Interceptor spec requires that we only throw Exception, NOT throwable, but
54          //runProfiledMethod throws Throwable.
55          try {
56              return runProfiledMethod(
57                      new AbstractJoinPoint() {
58                          public Object proceed() throws Throwable { return ctx.proceed(); }
59  
60                          public Object getExecutingObject() { return ctx.getTarget(); }
61  
62                          public Object[] getParameters() { return ctx.getParameters(); }
63  
64                          public String getMethodName() {
65                              return (executingMethod == null) ? "null" : executingMethod.getName();
66                          }
67                          
68                          public Class<?> getDeclaringClass() { return (executingMethod == null) ? null : executingMethod.getDeclaringClass() ; }
69                      },
70                      profiled,
71                      newStopWatch(profiled.logger(), profiled.level())
72              );
73          } catch (Exception e) {
74              throw e;
75          } catch (Error e) {
76              throw e;
77          } catch (Throwable t) {
78              //in practice this should rarely happen, as in standard usage Throwables fall under Exception or Error.
79              //However, by the Java spec, ONLY RuntimeExceptions and Errors are unchecked, so the compiler prevents us
80              //from executing "throw t" at this point. This seems like a poor design decision of the EJB spec, as all
81              //interceptors that wish to call a method that throws Throwable need to do something special like this.
82              //The best we can do here is wrap the Throwable as a RuntimeException, even though this could potentially
83              //change the semantics from the callers point of view.
84              throw new RuntimeException(t);
85          }
86      }
87  
88      /**
89       * Subclasses should implement this method to return a LoggingStopWatch that should be used to time the wrapped
90       * code block.
91       *
92       * @param loggerName The name of the logger to use for persisting StopWatch messages.
93       * @param levelName  The level at which the message should be logged.
94       * @return The new LoggingStopWatch.
95       */
96      protected abstract LoggingStopWatch newStopWatch(String loggerName, String levelName);
97  }
