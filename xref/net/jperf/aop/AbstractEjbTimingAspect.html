1   package net.jperf.aop;
2   
3   import net.jperf.LoggingStopWatch;
4   
5   import javax.interceptor.AroundInvoke;
6   import javax.interceptor.InvocationContext;
7   import java.lang.reflect.Method;
8   
9   /**
10   * This is the base class for TimingAspects that use the EJB interceptor framework.
11   * Subclasses just need to implement the {@link #newStopWatch} method to use their logging framework of choice
12   * (e.g. log4j or java.logging) to persist the StopWatch log message.
13   *
14   * @author Alex Devine
15   */
16  public abstract class AbstractEjbTimingAspect extends AgnosticTimingAspect {
17      /**
18       * This is the interceptor that runs the target method, surrounding it with stop watch start and stop calls.
19       *
20       * @param ctx The InvocationContext will be passed in by the Java EE server.
21       * @return The return value from the executed method.
22       * @throws Exception Any exceptions thrown by the executed method will bubble up.
23       */
24      @AroundInvoke
25      public Object doPerfLogging(final InvocationContext ctx) throws Exception {
26          final Method executingMethod = ctx.getMethod();
27  
28          //need to get the Profiled annotation off the method, otherwise use a default
29          Profiled profiled = (executingMethod == null) ?
30                              DefaultProfiled.INSTANCE :
31                              ctx.getMethod().getAnnotation(Profiled.class);
32          if (profiled == null) {
33              profiled = DefaultProfiled.INSTANCE;
34          }
35  
36          //note - the EJB 3.0 Interceptor spec requires that we only throw Exception, NOT throwable, but
37          //runProfiledMethod throws Throwable.
38          try {
39              return runProfiledMethod(
40                      new AbstractJoinPoint() {
41                          public Object proceed() throws Throwable { return ctx.proceed(); }
42  
43                          public Object getExecutingObject() { return ctx.getTarget(); }
44  
45                          public Object[] getParameters() { return ctx.getParameters(); }
46  
47                          public String getMethodName() {
48                              return (executingMethod == null) ? "null" : executingMethod.getName();
49                          }
50                          
51                          public Class<?> getDeclaringClass() { return (executingMethod == null) ? null : executingMethod.getDeclaringClass() ; }
52                      },
53                      profiled,
54                      newStopWatch(profiled.logger(), profiled.level())
55              );
56          } catch (Exception e) {
57              throw e;
58          } catch (Error e) {
59              throw e;
60          } catch (Throwable t) {
61              //in practice this should rarely happen, as in standard usage Throwables fall under Exception or Error.
62              //However, by the Java spec, ONLY RuntimeExceptions and Errors are unchecked, so the compiler prevents us
63              //from executing "throw t" at this point. This seems like a poor design decision of the EJB spec, as all
64              //interceptors that wish to call a method that throws Throwable need to do something special like this.
65              //The best we can do here is wrap the Throwable as a RuntimeException, even though this could potentially
66              //change the semantics from the callers point of view.
67              throw new RuntimeException(t);
68          }
69      }
70  
71      /**
72       * Subclasses should implement this method to return a LoggingStopWatch that should be used to time the wrapped
73       * code block.
74       *
75       * @param loggerName The name of the logger to use for persisting StopWatch messages.
76       * @param levelName  The level at which the message should be logged.
77       * @return The new LoggingStopWatch.
78       */
79      protected abstract LoggingStopWatch newStopWatch(String loggerName, String levelName);
80  }
