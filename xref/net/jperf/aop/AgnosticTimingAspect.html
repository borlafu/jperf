13  /**
14   * This AgnosticTimingAspect class contains all the logic for executing a profiled method with appropriate timing calls,
15   * but in an AOP-framework-agnostic way. You may choose to either extend or wrap this class to create an aspect in
16   * your desired framework. For example, if you look at the {@link AbstractTimingAspect}, you can see
17   * that it delegates all functionality to this class - it just includes the necessary AspectJ annotations and wraps
18   * the AspectJ-specific ProceedingJoinPoint as an {@link AbstractJoinPoint}.
19   *
20   * @author Marcin ZajÄ…czkowski, Alex Devine 
21   */
22  public class AgnosticTimingAspect {
23      /**
24       * This Map is used to cache compiled JEXL expressions. While theoretically unbounded, in reality the number of
25       * possible keys is equivalent to the number of unique JEXL expressions created in @Profiled annotations, which
26       * will have to be loaded in memory anyway when the class is loaded.
27       */
28      private Map<String, Expression> jexlExpressionCache = new ConcurrentHashMap<String, Expression>(64, .75F, 16);

30      /**
31       * This method actually executes the profiled method. Your AOP-framework-specific class should delegate to this
32       * method to proceed with execution.
33       *
34       * @param joinPoint The AOP join point - usually this will just be a simple wrapper around the
35       *                  AOP-framework-specific join point.
36       * @param profiled  The Profiled annotation that was set on the method being profiled.
37       * @param stopWatch This LogginStopWatch should be started JUST before this method is called.
38       * @return The return value from the profiled method.
39       * @throws Throwable Exception thrown by the profiled method will bubble up.
40       */
41      public Object runProfiledMethod(AbstractJoinPoint joinPoint, Profiled profiled, LoggingStopWatch stopWatch)
42              throws Throwable {
43          //if we're not going to end up logging the stopwatch, just run the wrapped method
44          if (!stopWatch.isLogging()) {
45              return joinPoint.proceed();
46          }
47  
48          stopWatch.setTimeThreshold(profiled.timeThreshold());
49          stopWatch.setNormalAndSlowSuffixesEnabled(profiled.normalAndSlowSuffixesEnabled());
50  
51          Object retVal = null;
52          Throwable exceptionThrown = null;
53          try {
54              return retVal = joinPoint.proceed();
55          } catch (Throwable t) {
56              throw exceptionThrown = t;
57          } finally {
58              String tag = getStopWatchTag(profiled, joinPoint, retVal, exceptionThrown);
59              String message = getStopWatchMessage(profiled, joinPoint, retVal, exceptionThrown);
60  
61              if (profiled.logFailuresSeparately()) {
62                  tag = (exceptionThrown == null) ? tag + ".success" : tag + ".failure";
63              }
64  
65              stopWatch.stop(tag, message);
66          }
67      }

69      /**
70       * Helper method gets the tag to use for StopWatch logging. Performs JEXL evaluation if necessary.
71       *
72       * @param profiled        The profiled annotation that was attached to the method.
73       * @param joinPoint       The AbstractJoinPoint encapulates the method around which this aspect advice runs.
74       * @param returnValue     The value returned from the execution of the profiled method, or null if the method
75       *                        returned void or an exception was thrown.
76       * @param exceptionThrown The exception thrown, if any, by the profiled method. Will be null if the method
77       *                        completed normally.
78       * @return The value to use as the StopWatch tag.
79       */
80      protected String getStopWatchTag(Profiled profiled, AbstractJoinPoint joinPoint, Object returnValue, Throwable exceptionThrown) {
81          String tag;
82          if (Profiled.DEFAULT_TAG_NAME.equals(profiled.tag())) {
83              // look for properties-based default
84              // if the tag name is not explicitly set on the Profiled annotation,
85              final String tagString = String.format("tag.%s.%s",
86                      joinPoint.getDeclaringClass().getName(),
87                      joinPoint.getMethodName());
88  
89              tag = Perf4jProperties.INSTANCE.getProperty(tagString);
90              String defaultTag = Perf4jProperties.INSTANCE.getProperty("defaultTag");
91  
92              if (tag == null && defaultTag == null) {
93                  // fall back to using the name of the method being annotated.
94                  tag = joinPoint.getMethodName();
95              }
96              else if (tag != null && tag.contains("{")) {
97                  tag = evaluateJexl(tag, joinPoint, returnValue, exceptionThrown);
98              }
99              else if (tag == null) {
100                 tag = evaluateJexl(defaultTag, joinPoint, returnValue, exceptionThrown);
101             }
102         } else if (profiled.el() && profiled.tag().contains("{")) {
103             tag = evaluateJexl(profiled.tag(), joinPoint, returnValue, exceptionThrown);
104         } else {
105             tag = profiled.tag();
106         }
107         return tag;
108     }

111     /**
112      * Helper method get the message to use for StopWatch logging. Performs JEXL evaluation if necessary.
113      *
114      * @param profiled        The profiled annotation that was attached to the method.
115      * @param joinPoint       The AbstractJoinPoint encapulates the method around which this aspect advice runs.
116      * @param returnValue     The value returned from the execution of the profiled method, or null if the method
117      *                        returned void or an exception was thrown.
118      * @param exceptionThrown The exception thrown, if any, by the profiled method. Will be null if the method
119      *                        completed normally.
120      * @return The value to use as the StopWatch message.
121      */
122     protected String getStopWatchMessage(Profiled profiled,
123                                          AbstractJoinPoint joinPoint,
124                                          Object returnValue,
125                                          Throwable exceptionThrown) {
126         String message;
127         if (profiled.message().length() == 0) {
128             // look for properties-based default
129             // if the message name is not explicitly set on the Profiled annotation,
130             String property = String.format("message.%s.%s", joinPoint.getDeclaringClass().getName(), joinPoint.getMethodName());
131             message = Perf4jProperties.INSTANCE.getProperty(property);
132 
133             if (message == null) {
134                 return null;
135             }
136             if (message.contains("{")) {
137                 message = evaluateJexl(message, joinPoint, returnValue, exceptionThrown);
138             }
139         } else if (profiled.el() && profiled.message().contains("{")) {
140             message = evaluateJexl(profiled.message(), joinPoint, returnValue, exceptionThrown);
141             if ("".equals(message)) {
142                 message = null;
143             }
144         } else {
145             message = profiled.message();
146         }
147         return message;
148     }

150     /**
151      * Helper method is used to parse out {expressionLanguage} elements from the text and evaluate the strings using
152      * JEXL.
153      *
154      * @param text            The text to be parsed.
155      * @param joinPoint       The AbstractJoinPoint encapulates the method around which this aspect advice runs.
156      * @param returnValue     The value returned from the execution of the profiled method, or null if the method
157      *                        returned void or an exception was thrown.
158      * @param exceptionThrown The exception thrown, if any, by the profiled method. Will be null if the method
159      *                        completed normally.
160      * @return The evaluated string.
161      * @see Profiled#el()
162      */
163 	protected String evaluateJexl(String text, AbstractJoinPoint joinPoint, Object returnValue, Throwable exceptionThrown) {
164         StringBuilder retVal = new StringBuilder(text.length());
165         JexlContext jexlContext = populateJexlContext(joinPoint, returnValue, exceptionThrown);
166 
167         // look for {expression} in the passed in text
168         int bracketIndex;
169         int lastCloseBracketIndex = -1;
170 
171         while ((bracketIndex = text.indexOf('{', lastCloseBracketIndex + 1)) >= 0) {
172             appendNonJexlText(retVal, text, bracketIndex, lastCloseBracketIndex);
173             lastCloseBracketIndex = lastCloseBracketIndex(text, bracketIndex);
174             String expressionText = textBetweenBrackets(text, bracketIndex, lastCloseBracketIndex);
175             evaluateJexlAndSetOnResult(retVal, expressionText, jexlContext);
176         }
177 
178         appendFinalPart(retVal, text, lastCloseBracketIndex);
179         return retVal.toString();
180     }
181 
182     private void appendFinalPart(StringBuilder retVal, String text, int lastCloseBracketIndex) {
183         if (lastCloseBracketIndex < text.length()) {
184             retVal.append(text.substring(lastCloseBracketIndex + 1, text.length()));
185         }
186     }
187 
188     private void evaluateJexlAndSetOnResult(StringBuilder retVal, String expressionText, JexlContext jexlContext) {
189         if (expressionText.length() == 0) {
190             return;
191         }
192         try {
193             Object result = getJexlExpression(expressionText).evaluate(jexlContext);
194             retVal.append(result);
195         } catch (Exception e) {
196             //we don't want to propagate exceptions up
197             retVal.append("_EL_ERROR_");
198         }
199     }
200 
201     private String textBetweenBrackets(String text, int bracketIndex, int lastCloseBracketIndex) {
202         return text.substring(bracketIndex + 1, lastCloseBracketIndex);
203     }
204 
205     private void appendNonJexlText(StringBuilder retVal, String text, int bracketIndex, int lastCloseBracketIndex) {
206         retVal.append(text.substring(lastCloseBracketIndex + 1, bracketIndex));
207     }
208 
209     private int lastCloseBracketIndex(String text, int bracketIndex) {
210         int index = text.indexOf('}', bracketIndex + 1);
211         if (index == -1) {
212             //if there wasn't a closing bracket index just go to the end of the string
213             index = text.length();
214         }
215 
216         return index;
217     }

219     @SuppressWarnings("unchecked")
220     private JexlContext populateJexlContext(AbstractJoinPoint joinPoint, Object returnValue, Throwable exceptionThrown) {
221         Object[] args = joinPoint.getParameters();
222 
223         //create a JexlContext to be used in all evaluations
224         JexlContext jexlContext = new HashMapContext();
225         for (int i = 0; i < args.length; i++) {
226             jexlContext.getVars().put("$" + i, args[i]);
227         }
228 
229         Class<?> declaringClass = joinPoint.getDeclaringClass();
230         jexlContext.getVars().put("$methodName", joinPoint.getMethodName());
231         jexlContext.getVars().put("$this", joinPoint.getExecutingObject());
232         jexlContext.getVars().put("$class", declaringClass);
233         jexlContext.getVars().put("$className", declaringClass.getName());
234         jexlContext.getVars().put("$classSimpleName", declaringClass.getSimpleName());
235         jexlContext.getVars().put("$return", returnValue);
236         jexlContext.getVars().put("$exception", exceptionThrown);
237 
238         return jexlContext;
239     }

241     /**
242      * Helper method gets a compiled JEXL expression for the specified expression text, either from the cache or by
243      * creating a new compiled expression.
244      *
245      * @param expressionText The JEXL expression text
246      * @return A compiled JEXL expression representing the expression text
247      * @throws Exception Thrown if there was an error compiling the expression text
248      */
249     protected Expression getJexlExpression(String expressionText) throws Exception {
250         Expression retVal = jexlExpressionCache.get(expressionText);
251         if (retVal == null) {
252             //Don't need synchronization here - if we end up calling createExpression in 2 separate threads, that's fine
253             jexlExpressionCache.put(expressionText, retVal = ExpressionFactory.createExpression(expressionText));
254         }
255         return retVal;
256     }
